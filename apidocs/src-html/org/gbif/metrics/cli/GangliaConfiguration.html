001package org.gbif.metrics.cli;
002
003import java.util.concurrent.TimeUnit;
004
005import com.beust.jcommander.Parameter;
006import org.codehaus.jackson.annotate.JsonIgnore;
007import com.google.common.base.Objects;
008import com.yammer.metrics.reporting.GangliaReporter;
009
010/**
011 * A configuration class which holds the host and port to connect yammer metrics to a ganglia server.
012 */
013@SuppressWarnings("PublicField")
014public class GangliaConfiguration {
015
016  @Parameter(names = "--ganglia-host")
017  public String gangliaHost;
018
019  @Parameter(names = "--ganglia-port")
020  public int gangliaPort;
021
022  /**
023   * Starts the GangliaReporter, pointing to the configured host and port.
024   */
025  @JsonIgnore
026  public void start() {
027    if (gangliaHost != null && gangliaPort > 0) {
028      GangliaReporter.enable(1, TimeUnit.MINUTES, gangliaHost, gangliaPort);
029    }
030  }
031
032  @Override
033  public String toString() {
034    return Objects.toStringHelper(this).add("gangliaHost", gangliaHost).add("gangliaPort", gangliaPort).toString();
035  }
036}
