001package org.gbif.metrics.cli;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.metrics.cube.HBaseCubes;
005import org.gbif.metrics.cube.tile.density.DensityCube;
006import org.gbif.metrics.cube.tile.density.DensityCubeUtil;
007import org.gbif.metrics.cube.tile.density.DensityCubeUtil.Op;
008import org.gbif.metrics.cube.tile.density.DensityTile;
009
010import java.io.IOException;
011
012import com.google.common.base.Throwables;
013import com.urbanairship.datacube.AsyncException;
014import com.urbanairship.datacube.Batch;
015import org.apache.hadoop.conf.Configuration;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * The callback for handling messages for density tile updates.
021 *
022 * NOTE: It is intended that only 1 be registered per queue.
023 *
024 * This configures an asynchronously flushing datacube using say N threads and a batch size of S.
025 * To ensure no contention, this will build a buffer of updates of size N*(S+1) before paritioning
026 * into N batches of size (S+1) and writing to DataCube.  By doing this, we know that datacube will
027 * flush them immediately.
028 *
029 * This is designed so that we can have numerous HBase flushing threads, which is required for any
030 * kind of performance, but ensure that they will never compete for updating cells.
031 *
032 * This class is threadsafe and stateful, but not intended to be used in multithreaded manner as it
033 * uses synchronized locks.
034 */
035public class DensityCubeCallback extends CubeUpdaterCallback<DensityTile> {
036  private static final Logger LOG = LoggerFactory.getLogger(DensityCubeCallback.class);
037  private final Batch<DensityTile> bufferBatch = new Batch<DensityTile>();
038  private final Object lock = new Object(); // for the batch
039  private final DensityCubeConfiguration configuration;
040  private final HBaseCubes<DensityTile> cube;
041
042  public DensityCubeCallback(DensityCubeConfiguration configuration, Configuration hadoopConfiguration) {
043    this.configuration = configuration;
044    try {
045      cube =
046        HBaseCubes.newCombiningBatchAsync(DensityCube.INSTANCE, DensityTile.DESERIALIZER,
047          configuration.cubeTable.getBytes(), configuration.columnFamily.getBytes(), hadoopConfiguration,
048          configuration.batchSize, configuration.batchFlushThreads, CubeUpdaterService.DEFAULT_IOE_RETRIES, Integer.MAX_VALUE);
049    } catch (IOException e) {
050      throw Throwables.propagate(e);
051    }
052  }
053
054  /**
055   * Closes the underlying datacube instance.
056   */
057  public void close() {
058    synchronized (lock) {
059      flushBatch(bufferBatch);
060    }
061    cube.closeQuietly();
062  }
063
064  /**
065   * Hands the update batch to datacube.
066   */
067  private void flushBatch(Batch<DensityTile> update) {
068    try {
069      if (update.getMap() != null && !update.getMap().isEmpty()) {
070        cube.write(update);
071      }
072    } catch (AsyncException e) {
073      LOG.error("AsyncException while updating cube with a batch size of {}", update.getMap().size(), e);
074    } catch (InterruptedException e) {
075      LOG.error("Interrupted while updating cube with a batch size of {}", update.getMap().size(), e);
076    }
077  }
078
079  @Override
080  HBaseCubes<DensityTile> getCube() {
081    return cube;
082  }
083
084  @Override
085  protected Batch<DensityTile> getNewMutations(Occurrence occurrence) {
086    return DensityCubeUtil.cubeMutations(occurrence, Op.ADDITION, configuration.zoom,
087      configuration.pixelsPerCluster);
088  }
089
090  @Override
091  protected Batch<DensityTile> getUpdateMutations(Occurrence oldOccurrence, Occurrence newOccurrence) {
092    Batch<DensityTile> update =
093      DensityCubeUtil.cubeMutations(oldOccurrence, Op.SUBTRACTION, configuration.zoom,
094        configuration.pixelsPerCluster);
095    if (newOccurrence != null) { // a delete has no new occurrence
096      update.putAll(DensityCubeUtil.cubeMutations(newOccurrence, Op.ADDITION, configuration.zoom,
097        configuration.pixelsPerCluster));
098    }
099    return update;
100  }
101}
