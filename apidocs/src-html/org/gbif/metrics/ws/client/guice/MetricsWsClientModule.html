001package org.gbif.metrics.ws.client.guice;
002
003import org.gbif.api.service.metrics.CubeService;
004import org.gbif.api.service.occurrence.OccurrenceCountryIndexService;
005import org.gbif.api.service.occurrence.OccurrenceDatasetIndexService;
006import org.gbif.api.service.occurrence.OccurrenceDistributionIndexService;
007import org.gbif.metrics.ws.client.CubeWsClient;
008import org.gbif.metrics.ws.client.OccurrenceCountryIndexWsClient;
009import org.gbif.metrics.ws.client.OccurrenceDatasetIndexWsClient;
010import org.gbif.metrics.ws.client.OccurrenceDistributionIndexWsClient;
011import org.gbif.ws.client.guice.GbifWsClientModule;
012
013import java.util.Properties;
014
015import com.google.inject.Provides;
016import com.google.inject.Scopes;
017import com.google.inject.Singleton;
018import com.google.inject.name.Named;
019import com.sun.jersey.api.client.Client;
020import com.sun.jersey.api.client.WebResource;
021
022/**
023 * Guice module that exposes the Metrics Service.
024 */
025public class MetricsWsClientModule extends GbifWsClientModule {
026
027  public MetricsWsClientModule(Properties properties) {
028    super(properties, CubeWsClient.class.getPackage());
029  }
030
031  /**
032   * @return the web resource to the webapp
033   */
034  @Provides
035  @Singleton
036  public WebResource provideWebResource(Client client, @Named("metrics.ws.url") String url) {
037    return client.resource(url);
038  }
039
040  @Override
041  protected void configureClient() {
042    bind(CubeService.class).to(CubeWsClient.class).in(Scopes.SINGLETON);
043    bind(OccurrenceDatasetIndexService.class).to(OccurrenceDatasetIndexWsClient.class).in(Scopes.SINGLETON);
044    bind(OccurrenceCountryIndexService.class).to(OccurrenceCountryIndexWsClient.class).in(Scopes.SINGLETON);
045    bind(OccurrenceDistributionIndexService.class).to(OccurrenceDistributionIndexWsClient.class).in(Scopes.SINGLETON);
046
047    expose(CubeService.class);
048    expose(OccurrenceDatasetIndexService.class);
049    expose(OccurrenceCountryIndexService.class);
050    expose(OccurrenceDistributionIndexService.class);
051  }
052}
