001package org.gbif.metrics.ws.client;
002
003import org.gbif.api.service.occurrence.OccurrenceDistributionIndexService;
004import org.gbif.api.vocabulary.BasisOfRecord;
005import org.gbif.api.vocabulary.Kingdom;
006import org.gbif.ws.client.BaseWsClient;
007
008import java.util.Map;
009
010import javax.validation.constraints.Min;
011import javax.ws.rs.core.MultivaluedMap;
012
013import com.google.common.base.Functions;
014import com.google.common.collect.ImmutableSortedMap;
015import com.google.common.collect.Ordering;
016import com.google.inject.Inject;
017import com.sun.jersey.api.client.GenericType;
018import com.sun.jersey.api.client.WebResource;
019import com.sun.jersey.core.util.MultivaluedMapImpl;
020
021/**
022 * Ws client for {@link OccurrenceDistributionIndexService}.
023 */
024public class OccurrenceDistributionIndexWsClient extends BaseWsClient implements OccurrenceDistributionIndexService {
025
026
027  private static final GenericType<Map<BasisOfRecord, Long>> BOF_MAP_GENERIC_TYPE =
028    new GenericType<Map<BasisOfRecord, Long>>() {
029    };
030
031  private static final GenericType<Map<Kingdom, Long>> KINGDOM_MAP_GENERIC_TYPE =
032    new GenericType<Map<Kingdom, Long>>() {
033    };
034
035  private static final GenericType<Map<Integer, Long>> INT_MAP_GENERIC_TYPE =
036    new GenericType<Map<Integer, Long>>() {
037    };
038
039
040  @Inject
041  public OccurrenceDistributionIndexWsClient(WebResource resource) {
042    super(resource);
043  }
044
045  @Override
046  public Map<BasisOfRecord, Long> getBasisOfRecordCounts() {
047    return getRequest("occurrence/counts/basis_of_record", BOF_MAP_GENERIC_TYPE);
048  }
049
050  @Override
051  public Map<Kingdom, Long> getKingdomCounts() {
052    return getRequest("occurrence/counts/kingdom", KINGDOM_MAP_GENERIC_TYPE);
053  }
054
055  @Override
056  public Map<Integer, Long> getYearCounts(@Min(0) int from, @Min(0) int to) {
057    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
058    params.putSingle("from", Integer.toString(from));
059    params.putSingle("to", Integer.toString(to));
060    return get(INT_MAP_GENERIC_TYPE, params, "occurrence/counts/year");
061  }
062
063  /**
064   * Executes a get request whose returned value is a SortedMap<Country, Integer>.
065   */
066  private <T extends Comparable<T>> Map<T, Long> getRequest(String path, GenericType<Map<T, Long>> genericType) {
067    final Map<T, Long> res = get(genericType, path);
068    return ImmutableSortedMap.copyOf(res,
069      Ordering.natural().onResultOf(Functions.forMap(res)).compound(Ordering.natural()).reverse());
070  }
071
072}
