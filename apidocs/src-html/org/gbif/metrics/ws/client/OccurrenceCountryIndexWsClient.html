004package org.gbif.metrics.ws.client;
005
006import org.gbif.api.service.occurrence.OccurrenceCountryIndexService;
007import org.gbif.api.vocabulary.Country;
008import org.gbif.ws.client.BaseWsClient;
009import org.gbif.ws.client.QueryParamBuilder;
010
011import java.util.Map;
012import javax.ws.rs.core.MultivaluedMap;
013
014import com.google.common.base.Functions;
015import com.google.common.collect.ImmutableSortedMap;
016import com.google.common.collect.Ordering;
017import com.google.inject.Inject;
018import com.sun.jersey.api.client.GenericType;
019import com.sun.jersey.api.client.WebResource;
020
021
022/**
023 * A web service client to support the accession of occurrence dataset indexes.
024 */
025public class OccurrenceCountryIndexWsClient extends BaseWsClient implements OccurrenceCountryIndexService {
026
027  private static final GenericType<Map<Country, Long>> GENERIC_TYPE = new GenericType<Map<Country, Long>>() {
028  };
029
030  @Inject
031  public OccurrenceCountryIndexWsClient(WebResource resource) {
032    super(resource);
033  }
034
035  @Override
036  public Map<Country, Long> publishingCountriesForCountry(Country country) {
037    return getRequest(QueryParamBuilder.create("country", country.getIso2LetterCode()).build(),
038      "occurrence/counts/publishing_countries");
039  }
040
041  @Override
042  public Map<Country, Long> countriesForPublishingCountry(Country publishingCountry) {
043    return getRequest(QueryParamBuilder.create("publishingCountry", publishingCountry.getIso2LetterCode()).build(),
044      "occurrence/counts/countries");
045  }
046
047  /**
048   * Executes a get request whose returned value is a SortedMap<Country, Integer>.
049   */
050  private Map<Country, Long> getRequest(MultivaluedMap<String, String> params, String path) {
051    final Map<Country, Long> res = get(GENERIC_TYPE, params, path);
052    return ImmutableSortedMap.copyOf(res,
053      Ordering.natural().onResultOf(Functions.forMap(res)).compound(Ordering.natural()).reverse());
054  }
055}
