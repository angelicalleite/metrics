001package org.gbif.metrics.ws.resources.provider;
002
003import org.gbif.api.exception.ServiceUnavailableException;
004import org.gbif.api.model.metrics.cube.Dimension;
005import org.gbif.metrics.cube.occurrence.OccurrenceCube;
006
007import java.util.List;
008import java.util.Map.Entry;
009import java.util.UUID;
010
011import javax.ws.rs.core.Context;
012import javax.ws.rs.core.MultivaluedMap;
013import javax.ws.rs.ext.Provider;
014
015import com.google.common.annotations.VisibleForTesting;
016import com.google.common.base.Preconditions;
017import com.google.common.collect.ImmutableMap;
018import com.google.inject.Singleton;
019import com.sun.jersey.api.core.HttpContext;
020import com.sun.jersey.api.model.Parameter;
021import com.sun.jersey.core.spi.component.ComponentContext;
022import com.sun.jersey.core.spi.component.ComponentScope;
023import com.sun.jersey.spi.inject.Injectable;
024import com.sun.jersey.spi.inject.InjectableProvider;
025import com.urbanairship.datacube.ReadBuilder;
026
027
028/**
029 * A provider that will convert parameters from the request into an INTERNAL cube reader.
030 */
031@Provider
032@Singleton
033public class OccurrenceCubeReaderProvider implements InjectableProvider<ProvidedOccurrenceCubeReader, Parameter> {
034
035  // Provides an index to lookup the API Dimension by the HTTP parameter key
036  private static final ImmutableMap<String, Dimension<?>> PARAM_INDEX;
037
038  static {
039    ImmutableMap.Builder<String, Dimension<?>> b = new ImmutableMap.Builder<String, Dimension<?>>();
040    for (Dimension<?> d : org.gbif.api.model.metrics.cube.OccurrenceCube.DIMENSIONS) {
041      b.put(d.getKey(), d);
042    }
043    PARAM_INDEX = b.build();
044  }
045
046  @Context
047  private final HttpContext hc;
048
049  public OccurrenceCubeReaderProvider(@Context HttpContext hc) {
050    this.hc = hc;
051  }
052
053  // The type casting is actually checked at runtime, and there _appears_ to be no way of
054  // checking the enum, hence the rawtypes and unchecked warnings suppression.
055  @SuppressWarnings({"unchecked", "rawtypes"})
056  @VisibleForTesting
057  ReadBuilder build(MultivaluedMap<String, String> params) throws ServiceUnavailableException, IllegalArgumentException {
058    ReadBuilder b = new ReadBuilder(OccurrenceCube.INSTANCE);
059    for (Entry<String, List<String>> param : params.entrySet()) {
060      // We only accept 1 value per parameter
061      String k = param.getKey();
062      String v = param.getValue().iterator().next();
063
064      // callback is a reserved word in the API and invokes the JS callback hack
065      // JQuery also inserts _ when using the callback=?
066      if ("callback".equalsIgnoreCase(k) || "_".equalsIgnoreCase(k)) {
067        continue;
068      }
069
070      Dimension<?> dim = PARAM_INDEX.get(k);
071      Preconditions.checkArgument(dim != null, "Unknown parameter found in the request, does not map to a dimension: "
072        + k);
073      Class<?> type = dim.getType();
074
075      // translate the API dimension to the internal dimension, typing the value
076      // Note: The API_MAPPING might be determined at runtime, with the cube's implementing an interface allowing that
077      // to be read.
078      com.urbanairship.datacube.Dimension<?> internalDim = OccurrenceCube.API_MAPPING.get(dim);
079      if (internalDim == null) {
080        throw new ServiceUnavailableException("Unable to convert address in the cube.  No dimension exists for "
081          + dim.getKey());
082      } else if (String.class == type) {
083        b.at(internalDim, v);
084
085      } else if (Integer.class == type) {
086        b.at(internalDim, Integer.valueOf(v));
087
088      } else if (Float.class == type) {
089        b.at(internalDim, Float.valueOf(v));
090
091      } else if (Double.class == type) {
092        b.at(internalDim, Double.valueOf(v));
093
094      } else if (Boolean.class == type) {
095        b.at(internalDim, Boolean.valueOf(v));
096
097      } else if (UUID.class == type) {
098        b.at(internalDim, UUID.fromString(v));
099
100      } else if (Enum.class.isAssignableFrom(type)) {
101        b.at(internalDim, Enum.valueOf((Class<Enum>) dim.getType(), v));
102
103      } else {
104        throw new ServiceUnavailableException("Dimension is of unknown type: " + type);
105      }
106    }
107    return b;
108  }
109
110  @Override
111  public Injectable<ReadBuilder> getInjectable(ComponentContext ic, ProvidedOccurrenceCubeReader a, Parameter c) {
112    return new Injectable<ReadBuilder>() {
113
114      @Override
115      public ReadBuilder getValue() {
116        return build(hc.getUriInfo().getQueryParameters());
117      }
118    };
119  }
120
121  @Override
122  public ComponentScope getScope() {
123    return ComponentScope.PerRequest;
124  }
125}
