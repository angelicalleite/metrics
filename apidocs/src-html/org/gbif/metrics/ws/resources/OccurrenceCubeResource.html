047/**
048 * A simple generic resource that will look up a numerical count from the named cube
049 * and address provided. Should no address be provided, a default builder which counts
050 * all records is used.
051 */
052@Path("/occurrence")
053@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
054public class OccurrenceCubeResource {
055
056  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceCubeResource.class);
057
058  private static Ordering<Map.Entry<?, Integer>> MAP_ORDER_DESC_INT = Ordering.natural()
059    .onResultOf(new Function<Map.Entry<?, Integer>, Integer>() {
060
061      public Integer apply(Map.Entry<?, Integer> entry) {
062        return entry.getValue();
063      }
064    }).reverse();
065
066  private static Ordering<Map.Entry<?, Long>> MAP_ORDER_DESC_LONG = Ordering.natural()
067    .onResultOf(new Function<Map.Entry<?, Long>, Long>() {
068
069      public Long apply(Map.Entry<?, Long> entry) {
070        return entry.getValue();
071      }
072    }).reverse();

074  // Since the cube does not support a count for all records, we do a basic sum
075  // Total = georeferenced + nonGeoferenced records
076  private static final Address GEOREFERENCED = new ReadBuilder(OccurrenceCube.INSTANCE)
077    .at(OccurrenceCube.IS_GEOREFERENCED, true).build();
078
079  private static final Address NOT_GEOREFERENCED = new ReadBuilder(OccurrenceCube.INSTANCE)
080    .at(OccurrenceCube.IS_GEOREFERENCED, false).build();

083  private final DataCubeIo<UuidIntMap> taxCubeIo;
084  private final DataCubeIo<UuidIntMap> countryCubeIo;

096  /**
097   * Looks up an addressable count from the cube, using the given {@link ReadBuilder}.
098   */
099  @GET
100  @Path("/count")
101  public Long count(@ProvidedOccurrenceCubeReader ReadBuilder b) {
102    Address a = b.build();
103    try {
104      // If no address, perform a SUM to get all
105      if (a.getBuckets().isEmpty()) {
106        return lookup(GEOREFERENCED) + lookup(NOT_GEOREFERENCED);
107      } else {
108        return lookup(a);
109      }
110    } catch (IllegalArgumentException e) {
111      throw new IllegalArgumentException("The provided address is not calculated in the cube: " + a);
112    } catch (Exception e) {
113      LOG.error("Unable to read from the cube", e);
114      throw new ServiceUnavailableException("Unable to read from the occurrence cube", e);
115    }
116  }

118  @GET
119  @Path("/counts/basis_of_record")
120  public Map<BasisOfRecord, Long> getBasisOfRecordCounts() {
121    Map<BasisOfRecord, Long> distribution = Maps.newHashMap();
122    try {
123      for (BasisOfRecord basisOfRecord : BasisOfRecord.values()) {
124        final Address address =
125          new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.BASIS_OF_RECORD, basisOfRecord).build();
126        distribution.put(basisOfRecord, lookup(address));
127      }
128    } catch (IllegalArgumentException e) {
129      throw new IllegalArgumentException("Error getting the occurrence counts", e);
130    } catch (Exception e) {
131      throw new ServiceUnavailableException("Error getting the occurrence counts", e);
132    }
133    return sortDescendingValues(distribution);
134  }

192  @GET
193  @Path("/counts/year")
194  public Map<Integer, Long> getYearCounts(@QueryParam("from") @Min(0) int from, @QueryParam("to") @Min(0) int to) {
195    Preconditions.checkArgument(from <= to, "Parameters are not a valid range of years");
196    ImmutableSortedMap.Builder<Integer, Long> distribution = ImmutableSortedMap.naturalOrder();
197    try {
198      for (int year = from; year < to; year++) {
199        final Address address =
200          new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.YEAR, year).build();
201        distribution.put(year, lookup(address));
202      }
203    } catch (IllegalArgumentException e) {
204      throw new IllegalArgumentException("Error getting the occurrence counts", e);
205    } catch (Exception e) {
206      throw new ServiceUnavailableException("Error getting the occurrence counts", e);
207    }
208    return distribution.build();
209  }

211  /**
212   * @return an immutable copy of the original, with the types casted and ordered
213   *         by the value descending
214   */
215  @VisibleForTesting
216  Map<UUID, Integer> sortDescending(Map<CharSequence, Integer> source) {
217    // Convert CharSequence to UUID map
218    Map<UUID, Integer> map = Maps.newHashMap();
219    for (Map.Entry<CharSequence, Integer> e : source.entrySet()) {
220      map.put(UUID.fromString(String.valueOf(e.getKey())), e.getValue());
221    }
222
223    return sort(map, MAP_ORDER_DESC_INT);
224  }
225
226  @VisibleForTesting
227  <K extends Comparable> Map<K, Long> sortDescendingValues(Map<K, Long> source) {
228    return sort(source, MAP_ORDER_DESC_LONG);
229  }
230
231  private <K extends Comparable<K>, V> Map<K, V> sort(Map<K, V> source, Ordering<Map.Entry<?, V>> ordering) {
232    ImmutableMap.Builder<K, V> builder = ImmutableMap.builder();
233    // we need a compund ordering to guarantee stable order with identical values
234
235    Ordering<Map.Entry<K, V>> keyOrder = Ordering.natural().onResultOf(new Function<Map.Entry<K, V>, K>() {
236        public K apply(Map.Entry<K, V> entry) {
237          return entry.getKey();
238        }
239      });
240
241    for (Map.Entry<K, V> entry : ordering.compound(keyOrder).sortedCopy(source.entrySet())) {
242      builder.put(entry.getKey(), entry.getValue());
243    }
244    return builder.build();
245  }

247  /**
248   * @param isPublishingCountry if given country parameter is a publishing country
249   */
250  private Map<Country, Long> getCountryMap(Country country, boolean isPublishingCountry) {
251    ReadBuilder b = new ReadBuilder(OccurrenceCube.INSTANCE);
252    if (isPublishingCountry) {
253      b.at(OccurrenceCube.HOST_COUNTRY, country);
254    } else {
255      b.at(OccurrenceCube.COUNTRY, country);
256    }
257
258    Map<Country, Long> map = Maps.newHashMap();
259    // lacking a specific cube we iterate over all official countries to generate a sorted map
260    try {
261      // we reuse the same address instance with changing dimension value below
262      // the read builder is attached to the same instance all the time and only the latest value for a given dimension
263      // is kept
264      Address address = b.build();
265      for (Country c : Country.values()) {
266        if (c.isOfficial()) {
267          if (isPublishingCountry) {
268            b.at(OccurrenceCube.COUNTRY, c);
269          } else {
270            b.at(OccurrenceCube.HOST_COUNTRY, c);
271          }
272
273          long cnt = lookup(address);
274          if (cnt > 0) {
275            map.put(c, cnt);
276          }
277        }
278      }
279    } catch (IOException e) {
280      throw new ServiceUnavailableException("Unable to read from the occurrence cube", e);
281    } catch (InterruptedException e) {
282      throw new ServiceUnavailableException("Unable to read from the occurrence cube", e);
283    }
284
285    return sortDescendingValues(map);
286  }

297  /**
298   * Gets the a sorted Map<UUID, Integer> UuidIntMap from a DataCube<UuidIntMap>.
299   */
300  private <O> Map<UUID, Integer> getMapFromCube(DataCubeIo<UuidIntMap> dataCubeIo, DataCube<UuidIntMap> dataCube,
301    Dimension<?> dimension, O coordinate) {
302    try {
303      Optional<UuidIntMap> o = null;
304      if (coordinate != null) {
305        o = dataCubeIo.get(new ReadBuilder(dataCube).at(dimension, coordinate));
306      }
307      if (o != null && o.isPresent()) {
308        return sortDescending(o.get().getCounts());
309      } else {
310        return sortDescending(UuidIntMap.EMPTY_MAP.getCounts());
311      }
312    } catch (Exception e) {
313      e.printStackTrace();
314      LOG.error("Unable to read from the cube", e);
315      throw new ServiceUnavailableException("Unable to read from the taxon occurrence dataset cube", e);
316    }
317  }

319  /**
320   * Simple lookup utility
321   * 
322   * @param a To lookup
323   * @return The count, which may be calculated to be 0
324   * @throws IOException On communication to the cube (typically an HBase layer)
325   * @throws InterruptedException If something nasty happens within the cube (e.g.) some fatal service error
326   * @throws IllegalArgumentException Should the provided Address NOT be calculated in the cube
327   */
328  private long lookup(Address a) throws IOException, InterruptedException, IllegalArgumentException {
329    Optional<LongOp> o = occCubeIo.get(a);
330    if (o.isPresent()) {
331      return o.get().getLong();
332    } else {
333      return 0L;
334    }
335  }
