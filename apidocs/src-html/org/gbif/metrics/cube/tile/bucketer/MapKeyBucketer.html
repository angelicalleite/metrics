001package org.gbif.metrics.cube.tile.bucketer;
002
003import com.google.common.base.Strings;
004import com.urbanairship.datacube.CSerializable;
005import com.urbanairship.datacube.bucketers.AbstractIdentityBucketer;
006import com.urbanairship.datacube.serializables.StringSerializable;
007
008/**
009 * Encodes and decodes the various map keys into 36 character bytes padding with spaces.
010 * 36 characters are required for UUIDs which are considered the longest keys supported by maps.
011 */
012public class MapKeyBucketer extends AbstractIdentityBucketer<String> {
013  private static final MapKeyBucketer instance = new MapKeyBucketer();
014  public static final int BYTES=36;
015
016  @Override
017  public CSerializable makeSerializable(String coordinateField) {
018    return new StringSerializable(Strings.padStart(coordinateField, 36, ' '));
019  }
020
021  public static final MapKeyBucketer getInstance() {
022    return instance;
023  }
024}
