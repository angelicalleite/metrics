001package org.gbif.metrics.cube.tile.point;
002
003import java.util.List;
004
005import com.google.common.collect.ImmutableList;
006import com.urbanairship.datacube.DataCube;
007import com.urbanairship.datacube.Dimension;
008import com.urbanairship.datacube.Rollup;
009import com.urbanairship.datacube.bucketers.BigEndianIntBucketer;
010
011/**
012 * The cube definition for a point based map.
013 */
014public class PointCube {
015
016  // no id substitution, 4 bytes for int
017  public static final Dimension<Integer> TAXON_ID = new Dimension<Integer>("taxonID", new BigEndianIntBucketer(), false, 4);
018  public static final Dimension<Integer> ZOOM = new Dimension<Integer>("zoom", new BigEndianIntBucketer(), false, 4);
019  public static final Dimension<Integer> TILE_X = new Dimension<Integer>("tileX", new BigEndianIntBucketer(), false, 4);
020  public static final Dimension<Integer> TILE_Y = new Dimension<Integer>("tileY", new BigEndianIntBucketer(), false, 4);
021
022  // Singleton instance if accessed through the instance() method
023  public static final DataCube<PointTile> INSTANCE = newInstance();
024
025  // Not for instantiation
026  private PointCube() {
027  }
028
029  private static DataCube<PointTile> newInstance() {
030    List<Dimension<?>> dimensions = ImmutableList.<Dimension<?>>of(TAXON_ID, ZOOM, TILE_X, TILE_Y);
031    List<Rollup> rollups = ImmutableList.of(new Rollup(TAXON_ID, ZOOM, TILE_X, TILE_Y));
032    return new DataCube<PointTile>(dimensions, rollups);
033  }
034}
