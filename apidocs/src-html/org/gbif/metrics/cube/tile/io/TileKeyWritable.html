001package org.gbif.metrics.cube.tile.io;
002
003import java.io.DataInput;
004import java.io.DataOutput;
005import java.io.IOException;
006
007import com.google.common.base.Objects;
008import org.apache.hadoop.io.WritableComparable;
009
010/**
011 * A container object intended for use as MR keys to encapsulate everything needed to identify
012 * a tile.
013 */
014public class TileKeyWritable implements WritableComparable<TileKeyWritable> {
015
016  private TileContentType contentType;
017  private String key;
018  private int x, y, z;
019
020  public TileKeyWritable() {
021  }
022
023  public TileKeyWritable(TileContentType type, String key, int x, int y, int z) {
024    this.contentType = type;
025    this.key = key;
026    this.x = x;
027    this.y = y;
028    this.z = z;
029  }
030
031  @Override
032  public int compareTo(TileKeyWritable o) {
033    return this.toString().compareTo(o.toString());
034  }
035
036  @Override
037  public boolean equals(Object object) {
038    if (this == object) {
039      return true;
040    }
041    if (object instanceof TileKeyWritable) {
042      TileKeyWritable that = (TileKeyWritable) object;
043      return Objects.equal(this.contentType, that.contentType) && Objects.equal(this.key, that.key) && Objects.equal(this.x, that.x)
044        && Objects.equal(this.y, that.y) && Objects.equal(this.z, that.z);
045    }
046    return false;
047  }
048
049  public String getKey() {
050    return key;
051  }
052
053  public TileContentType getType() {
054    return contentType;
055  }
056
057  public int getX() {
058    return x;
059  }
060
061  public int getY() {
062    return y;
063  }
064
065  public int getZ() {
066    return z;
067  }
068
069  @Override
070  public int hashCode() {
071    // Note the getId() since enumeration hashcodes are INCONSISTENT across JVMs
072    return Objects.hashCode(contentType.getId(), key, x, y, z);
073  }
074
075  @Override
076  public void readFields(DataInput in) throws IOException {
077    contentType = TileContentType.INSTANCE(in.readInt());
078    key = in.readUTF();
079    x = in.readInt();
080    y = in.readInt();
081    z = in.readInt();
082  }
083
084  public void setKey(String key) {
085    this.key = key;
086  }
087
088
089  public void setType(TileContentType type) {
090    this.contentType = type;
091  }
092
093  public void setX(int x) {
094    this.x = x;
095  }
096
097  public void setY(int y) {
098    this.y = y;
099  }
100
101  public void setZ(int z) {
102    this.z = z;
103  }
104
105  @Override
106  public String toString() {
107    return Objects.toStringHelper(this).add("contentType", contentType).add("key", key).add("x", x).add("y", y).add("z", z).toString();
108  }
109
110  @Override
111  public void write(DataOutput out) throws IOException {
112    out.writeInt(contentType.getId());
113    out.writeUTF(key);
114    out.writeInt(x);
115    out.writeInt(y);
116    out.writeInt(z);
117  }
118}
