001package org.gbif.metrics.cube.tile.io;
002
003import java.io.DataInput;
004import java.io.DataOutput;
005import java.io.IOException;
006
007import com.google.common.base.Objects;
008import org.apache.hadoop.io.WritableComparable;
009
010/**
011 * A container object intended for use as MR values to encapsulate a latitude
012 * and longitude pair.
013 */
014public class LatLngWritable implements WritableComparable<LatLngWritable> {
015
016  private double lat, lng;
017  private int count;
018
019  public LatLngWritable() {
020  }
021
022  public LatLngWritable(double lat, double lng, int count) {
023    this.lat = lat;
024    this.lng = lng;
025    this.count = count;
026  }
027
028  @Override
029  public int compareTo(LatLngWritable o) {
030    return this.toString().compareTo(o.toString());
031  }
032
033  @Override
034  public boolean equals(Object object) {
035    if (object instanceof LatLngWritable) {
036      LatLngWritable that = (LatLngWritable) object;
037      return Objects.equal(this.lat, that.lat) && Objects.equal(this.lng, that.lng) && Objects.equal(this.count, that.count);
038    }
039    return false;
040  }
041
042  public int getCount() {
043    return count;
044  }
045
046  public double getLat() {
047    return lat;
048  }
049
050  public double getLng() {
051    return lng;
052  }
053
054  @Override
055  public int hashCode() {
056    return Objects.hashCode(lat, lng, count);
057  }
058
059  @Override
060  public void readFields(DataInput in) throws IOException {
061    lat = in.readDouble();
062    lng = in.readDouble();
063    count = in.readInt();
064  }
065
066  public void setCount(int count) {
067    this.count = count;
068  }
069
070  public void setLat(double lat) {
071    this.lat = lat;
072  }
073
074  public void setLng(double lng) {
075    this.lng = lng;
076  }
077
078  @Override
079  public String toString() {
080    return Objects.toStringHelper(this).add("lat", lat).add("lng", lng).add("count", count).toString();
081  }
082
083  @Override
084  public void write(DataOutput out) throws IOException {
085    out.writeDouble(lat);
086    out.writeDouble(lng);
087    out.writeInt(count);
088  }
089}
