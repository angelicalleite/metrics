001package org.gbif.metrics.cube.tile;
002
003import java.awt.Point;
004import java.awt.geom.Point2D;
005import java.awt.geom.Rectangle2D;
006
007/**
008 * Utilities for dealing with Google tiles.
009 */
010public class MercatorProjectionUtil {
011
012  public static final int TILE_SIZE = 256;
013
014  /**
015   * @return the X pixel offset within the tile
016   */
017  public static int getOffsetX(double lat, double lng, int zoom) {
018    double x = toNormalisedPixelCoords(lat, lng).getX();
019    int scale = 1 << zoom;
020    x *= scale * TILE_SIZE;
021    return (int) (x % TILE_SIZE);
022  }
023
024
025  /**
026   * @return the Y pixel offset within the tile.
027   */
028  public static int getOffsetY(double lat, double lng, int zoom) {
029    double y = toNormalisedPixelCoords(lat, lng).getY();
030    int scale = 1 << zoom;
031    y *= scale * TILE_SIZE;
032    int tileY = toTileY(lat, zoom);
033    return (int) (y - (TILE_SIZE * tileY));
034  }
035
036  /**
037   * @return a Rectangle2D with x = lon, y = lat, width=lonSpan, height=latSpan
038   *         for an x,y,zoom as used by google.
039   */
040  public static Rectangle2D.Double getTileRect(int x, int y, int zoom) {
041    int tilesAtThisZoom = 1 << (17 - zoom);
042    double lngWidth = 360.0 / tilesAtThisZoom; // width in degrees longitude
043    double lng = -180 + (x * lngWidth); // left edge in degrees longitude
044
045    double latHeightMerc = 1.0 / tilesAtThisZoom; // height in "normalized" mercator 0,0 top left
046    double topLatMerc = y * latHeightMerc; // top edge in "normalized" mercator 0,0 top left
047    double bottomLatMerc = topLatMerc + latHeightMerc;
048
049    // convert top and bottom lat in mercator to degrees
050    // note that in fact the coordinates go from about -85 to +85 not -90 to 90!
051    double bottomLat = Math.toDegrees((2 * Math.atan(Math.exp(Math.PI * (1 - (2 * bottomLatMerc))))) - (Math.PI / 2));
052
053    double topLat = Math.toDegrees((2 * Math.atan(Math.exp(Math.PI * (1 - (2 * topLatMerc))))) - (Math.PI / 2));
054
055    double latHeight = topLat - bottomLat;
056
057    return new Rectangle2D.Double(lng, bottomLat, lngWidth, latHeight);
058  }
059
060  /**
061   * Google maps cover +/- 85 degrees only.
062   * 
063   * @return true if the location is plottable on a map
064   */
065  public static boolean isPlottable(Double lat, Double lng) {
066    return (lat != null && lng != null && lat >= -85d && lat <= 85d && lng >= -180 && lng <= 180);
067  }
068
069  /**
070   * Returns the lat/lng as an "Offset Normalized Mercator" pixel coordinate,
071   * this is a coordinate that runs from 0..1 in latitude and longitude with 0,0 being
072   * top left. Normalizing means that this routine can be used at any zoom level and
073   * then multiplied by a power of two to get actual pixel coordinates.
074   */
075  public static Point2D toNormalisedPixelCoords(double lat, double lng) {
076    // first convert to Mercator projection
077    // first convert the lat lon to mercator coordintes.
078    if (lng > 180) {
079      lng -= 360;
080    }
081
082    lng /= 360;
083    lng += 0.5;
084
085    lat = 0.5 - ((Math.log(Math.tan((Math.PI / 4) + ((0.5 * Math.PI * lat) / 180))) / Math.PI) / 2.0);
086
087    return new Point2D.Double(lng, lat);
088  }
089
090  /**
091   * @return The tile index for the longitude at the given zoom
092   */
093  public static int toTileX(double lng, int zoom) {
094    if (lng > 180) {
095      lng -= 360;
096    }
097
098    lng /= 360;
099    lng += 0.5;
100    int scale = 1 << zoom;
101    return (int) (lng * scale);
102  }
103
104
105  /**
106   * Returns a point that is a google tile reference for the tile containing
107   * the lat/lng and at the zoom level.
108   */
109  public static Point toTileXY(double lat, double lng, int zoom) {
110    Point2D normalised = toNormalisedPixelCoords(lat, lng);
111    int scale = 1 << zoom;
112
113    // can just truncate to integer, this looses the fractional "pixel offset"
114    return new Point((int) (normalised.getX() * scale), (int) (normalised.getY() * scale));
115  }
116
117  /**
118   * @return The tile index for the latitude at the given zoom
119   */
120  public static int toTileY(double lat, int zoom) {
121    lat = 0.5 - ((Math.log(Math.tan((Math.PI / 4) + ((0.5 * Math.PI * lat) / 180))) / Math.PI) / 2.0);
122    int scale = 1 << zoom;
123    return (int) (lat * scale);
124  }
125}
