001package org.gbif.metrics.cube.tile.density.io;
002
003import org.gbif.metrics.cube.util.SerDeUtils;
004
005import java.io.IOException;
006
007import org.apache.avro.generic.GenericDatumWriter;
008import org.apache.avro.specific.SpecificDatumReader;
009import org.apache.avro.specific.SpecificDatumWriter;
010
011public class DensityTileAvroSerDeUtils extends SerDeUtils {
012
013  private static final GenericDatumWriter<DensityTileAvro> WRITER = new SpecificDatumWriter<DensityTileAvro>(DensityTileAvro.class);
014  private static final SpecificDatumReader<DensityTileAvro> READER = new SpecificDatumReader<DensityTileAvro>(DensityTileAvro.class);
015
016  public static DensityTileAvro decode(final byte[] data) throws IOException {
017    return decodeObject(new DensityTileAvro(), data, READER);
018  }
019
020  public static byte[] encode(final DensityTileAvro tile) throws IOException {
021    return encodeObject(tile, WRITER);
022  }
023}
