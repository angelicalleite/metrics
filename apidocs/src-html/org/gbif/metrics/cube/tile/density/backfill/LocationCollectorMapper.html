001package org.gbif.metrics.cube.tile.density.backfill;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.api.vocabulary.EndpointType;
005import org.gbif.metrics.cube.mapred.OccurrenceWritable;
006import org.gbif.metrics.cube.tile.MercatorProjectionUtil;
007import org.gbif.occurrencestore.common.model.constants.FieldName;
008import org.gbif.occurrencestore.persistence.OccurrenceResultReader;
009import org.gbif.occurrencestore.util.BasisOfRecordConverter;
010
011import java.io.IOException;
012
013import org.apache.hadoop.hbase.client.Result;
014import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
015import org.apache.hadoop.hbase.mapreduce.TableMapper;
016import org.apache.hadoop.io.IntWritable;
017
018/**
019 * Reads the HBase table, collecting the terms we are interested in as we know there are duplicates
020 * at the same location. This is only to reduce the amount of grouping needed in later MR jobs.
021 */
022public class LocationCollectorMapper extends TableMapper<OccurrenceWritable, IntWritable> {
023
024  private final static IntWritable ONE = new IntWritable(1);
025  private final BasisOfRecordConverter borc = new BasisOfRecordConverter();
026
027  /**
028   * Reads the table, emits the Result keyed on the Lat Lng if it is a plottable record
029   */
030  @Override
031  protected void map(ImmutableBytesWritable key, Result row, Context context) throws IOException, InterruptedException {
032
033    Double latitude = OccurrenceResultReader.getDouble(row, FieldName.I_LATITUDE);
034    Double longitude = OccurrenceResultReader.getDouble(row, FieldName.I_LONGITUDE);
035
036    // Google only goes +/- 85 degrees and we only want maps with no known issues
037    if (!OccurrenceWritable.hasSpatialIssue(row) && MercatorProjectionUtil.isPlottable(latitude, longitude)) {
038
039      // Note: Make sure everything read here is in the getScanner() in BackFillCallback!
040      Integer kingdomID = OccurrenceResultReader.getInteger(row, FieldName.I_KINGDOM_ID);
041      Integer phylumID = OccurrenceResultReader.getInteger(row, FieldName.I_PHYLUM_ID);
042      Integer classID = OccurrenceResultReader.getInteger(row, FieldName.I_CLASS_ID);
043      Integer orderID = OccurrenceResultReader.getInteger(row, FieldName.I_ORDER_ID);
044      Integer familyID = OccurrenceResultReader.getInteger(row, FieldName.I_FAMILY_ID);
045      Integer genusID = OccurrenceResultReader.getInteger(row, FieldName.I_GENUS_ID);
046      Integer speciesID = OccurrenceResultReader.getInteger(row, FieldName.I_SPECIES_ID);
047      // taxon != species (it may be a higher taxon, or might be a subspecies)
048      Integer taxonID = OccurrenceResultReader.getInteger(row, FieldName.I_NUB_ID);
049      String publishingOrganisationKey = OccurrenceResultReader.getString(row, FieldName.OWNING_ORG_KEY);
050      String datasetKey = OccurrenceResultReader.getString(row, FieldName.DATASET_KEY);
051      String countryIsoCode = OccurrenceResultReader.getString(row, FieldName.I_ISO_COUNTRY_CODE);
052      String hostCountryIsoCode = OccurrenceResultReader.getString(row, FieldName.HOST_COUNTRY);
053      Integer year = OccurrenceResultReader.getInteger(row, FieldName.I_YEAR);
054
055      Integer borAsInt = OccurrenceResultReader.getInteger(row, FieldName.I_BASIS_OF_RECORD);
056      BasisOfRecord bor = borc.toEnum(borAsInt);
057      EndpointType protocol = EndpointType.valueOf(OccurrenceResultReader.getString(row, FieldName.PROTOCOL));
058
059      context.getCounter(bor).increment(1);
060
061      context.write(new OccurrenceWritable(kingdomID, phylumID, classID, orderID, familyID, genusID, speciesID,
062        taxonID, false, publishingOrganisationKey, datasetKey, countryIsoCode, hostCountryIsoCode, latitude,
063        longitude, year, null, // month
064        bor, protocol, 1), ONE);
065    }
066  }
067}
