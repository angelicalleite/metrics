001package org.gbif.metrics.cube.tile.density.backfill;
002
003import org.gbif.metrics.cube.HBaseSourcedBackfill;
004import org.gbif.metrics.cube.tile.density.DensityTile;
005import org.gbif.utils.file.properties.PropertiesUtil;
006
007import java.io.IOException;
008
009import org.apache.hadoop.conf.Configuration;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013
014/**
015 * Runs a backfill to populate the tile densities.
016 * TODO: remove this as per http://dev.gbif.org/issues/browse/POR-394
017 */
018public class Backfill extends HBaseSourcedBackfill {
019
020  private static final Logger LOG = LoggerFactory.getLogger(Backfill.class);
021
022  private final int numZooms;
023  private final int pixelsPerCluster;
024
025  // sensible defaults when omitted
026  final static int DEFAULT_PIXELS_PER_CLUSTER = 4;
027  final static int DEFAULT_NUM_ZOOMS = 4;
028
029  public static final String KEY_NUM_ZOOMS = "density-cube.numZooms";
030  public static final String KEY_PIXELS_PER_CLUSTER = "density-cube.tilePixelsPerCluster";
031
032  // the prefix used in the cube.properties
033  private static final String PREFIX = "density-cube";
034
035  public Backfill() throws IOException {
036    super(PREFIX);
037    pixelsPerCluster = PropertiesUtil.propertyAsInt(getProperties(), KEY_PIXELS_PER_CLUSTER, false, DEFAULT_PIXELS_PER_CLUSTER);
038    numZooms = PropertiesUtil.propertyAsInt(getProperties(), KEY_NUM_ZOOMS, false, DEFAULT_NUM_ZOOMS);
039  }
040
041  public static void main(String[] args) {
042    try {
043      Backfill app = new Backfill();
044      app.backfill(new BackfillCallback(), DensityTile.DensityTileDeserializer.class);
045    } catch (Exception e) {
046      LOG.error("Error running backfill", e);
047    }
048  }
049
050  /**
051   * Adds more values to the hadoop configuration.
052   */
053  @Override
054  protected void setup(Configuration conf) throws IOException {
055    super.setup(conf);
056    conf.setInt(KEY_PIXELS_PER_CLUSTER, pixelsPerCluster);
057    conf.setInt(KEY_NUM_ZOOMS, numZooms);
058  }
059}
