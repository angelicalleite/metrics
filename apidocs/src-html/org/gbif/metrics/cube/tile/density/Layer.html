004package org.gbif.metrics.cube.tile.density;
005
006import org.gbif.api.vocabulary.BasisOfRecord;
007import org.gbif.metrics.cube.mapred.OccurrenceWritable;
008
009
010/**
011 * Defines the layers (grids) that a density tile may hold.
012 * [Future selves of us might question why we only go up to 2020, but unlikely we will use this code base in 2020.]
013 */
014public enum Layer {
015  // Specimens
016  SP_NO_YEAR, SP_PRE_1900, SP_1900_1910, SP_1910_1920, SP_1920_1930, SP_1930_1940, SP_1940_1950, SP_1950_1960,
017  SP_1960_1970, SP_1970_1980, SP_1980_1990, SP_1990_2000, SP_2000_2010,
018  SP_2010_2020,
019
020  // Observations
021  OBS_NO_YEAR, OBS_PRE_1900, OBS_1900_1910, OBS_1910_1920, OBS_1920_1930, OBS_1930_1940, OBS_1940_1950, OBS_1950_1960,
022  OBS_1960_1970, OBS_1970_1980, OBS_1980_1990, OBS_1990_2000, OBS_2000_2010, OBS_2010_2020,
023
024  // Living (data does not support time series)
025  LIVING,
026
027  // Fossils (data does not support time series)
028  FOSSIL,
029
030  // Other basis of record catches a lot of things (so year makes sense still)
031  OTH_NO_YEAR, OTH_PRE_1900, OTH_1900_1910, OTH_1910_1920, OTH_1920_1930, OTH_1930_1940, OTH_1940_1950, OTH_1950_1960,
032  OTH_1960_1970, OTH_1970_1980, OTH_1980_1990, OTH_1990_2000, OTH_2000_2010, OTH_2010_2020;
033
034  // utility arrays used in inference only.
035  // arrays ordering is tightly coupled to the inference code for the year below.
036  private final static Layer[] SPECIMENS = {SP_NO_YEAR, SP_PRE_1900, SP_1900_1910, SP_1910_1920, SP_1920_1930,
037    SP_1930_1940, SP_1940_1950, SP_1950_1960, SP_1960_1970, SP_1970_1980, SP_1980_1990, SP_1990_2000, SP_2000_2010,
038    SP_2010_2020};
039  private final static Layer[] OBSERVATIONS = {OBS_NO_YEAR, OBS_PRE_1900, OBS_1900_1910, OBS_1910_1920, OBS_1920_1930,
040    OBS_1930_1940, OBS_1940_1950, OBS_1950_1960, OBS_1960_1970, OBS_1970_1980, OBS_1980_1990, OBS_1990_2000,
041    OBS_2000_2010, OBS_2010_2020};
042  private final static Layer[] OTHERS = {OTH_NO_YEAR, OTH_PRE_1900, OTH_1900_1910, OTH_1910_1920, OTH_1920_1930,
043    OTH_1930_1940, OTH_1940_1950, OTH_1950_1960, OTH_1960_1970, OTH_1970_1980, OTH_1980_1990, OTH_1990_2000,
044    OTH_2000_2010, OTH_2010_2020};
045
046  /**
047   * Utility to determine to which layer an occurrence record belongs.
048   * This assumes that the record is plottable, so no verification on the coordinates is performed.
049   * 
050   * @param bor The basis of record for the Occurrence
051   * @param year The year for the Occurrence
052   * @return The layer that a record should belong to.
053   */
054  public static Layer inferFrom(BasisOfRecord bor, Integer year) {
055    switch (bor) {
056      case FOSSIL_SPECIMEN:
057        return FOSSIL;
058      case LIVING_SPECIMEN:
059        return LIVING;
060
061      case PRESERVED_SPECIMEN:
062        return inferYear(year, SPECIMENS);
063
064      case OBSERVATION:
065      case HUMAN_OBSERVATION:
066      case MACHINE_OBSERVATION:
067        return inferYear(year, OBSERVATIONS);
068
069      default:
070        return inferYear(year, OTHERS);
071    }
072  }
073
074  /**
075   * Utility to determine to which layer an occurrence record belongs.
076   * This assumes that the record is plottable, so no verification on the coordinates is performed.
077   * 
078   * @param o The Occurrence record which needs to be placed.
079   * @return The layer that a record should belong to.
080   */
081  public static Layer inferFrom(OccurrenceWritable o) {
082    return inferFrom(o.getBasisOfRecord(), o.getYear());
083  }
084
085
086  /**
087   * Given the choices which are of a careful ordering (see definitions above) returns the
088   * correct year.
089   */
090  private static Layer inferYear(Integer year, Layer[] choices) {
091    if (year == null) {
092      return choices[0];
093    }
094
095    if (year < 1900) {
096      return choices[1];
097    }
098
099    int index = 2;
100    for (int decade = 1900; decade <= 2010; decade += 10, index += 1) {
101      if (year >= decade && year < decade + 10) {
102        return choices[index];
103      }
104    }
105
106    // could be post 2020 for example
107    return choices[0];
108  }
109}
