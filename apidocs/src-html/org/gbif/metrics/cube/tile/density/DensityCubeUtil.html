001package org.gbif.metrics.cube.tile.density;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.metrics.cube.tile.MercatorProjectionUtil;
005import org.gbif.metrics.cube.tile.io.TileContentType;
006
007import java.util.Set;
008
009import com.google.common.collect.Sets;
010import com.urbanairship.datacube.Batch;
011import com.urbanairship.datacube.WriteBuilder;
012
013/**
014 * A utility class to encapsulate the logic needed to convert between the denormalized
015 * format of occurrences (which have kingdomKey, phylumKey etc) and the mutations that need to happen to the tiles.
016 */
017public class DensityCubeUtil {
018  public static enum Op {
019    ADDITION, SUBTRACTION
020  }
021
022  /**
023   * Utility to either add or subtract one, to the batch for the given record.
024   */
025  private static void addMutations(Batch<DensityTile> batch, TileContentType type, String key, int x, int y, int z,
026    int pixelsPerCluster, Op op, Occurrence o) {
027    DensityTile.Builder b = DensityTile.builder(z, x, y, pixelsPerCluster);
028    int count = (op == Op.SUBTRACTION) ? -1 : 1;
029    b.collect(Layer.inferFrom(o.getBasisOfRecord(), o.getYear()), o.getLatitude(), o.getLongitude(), count);
030    DensityTile tile = b.build();
031    WriteBuilder wb =
032      new WriteBuilder(DensityCube.INSTANCE).at(DensityCube.ZOOM, z).at(DensityCube.TILE_X, x)
033        .at(DensityCube.TILE_Y, y).at(DensityCube.KEY, key).at(DensityCube.TYPE, type);
034    batch.putAll(DensityCube.INSTANCE.getWrites(wb, tile));
035  }
036
037  /**
038   * For the given occurrence, determines the mutations (addresses and operations) that need
039   * to be applied.
040   *
041   * @param o The denormalized representation
042   * @param op That is going to be applied to the cube
043   * @return The batch of updates to apply
044   */
045  public static Batch<DensityTile> cubeMutations(Occurrence o, Op op, int zoom, int pixelsPerCluster) {
046    Double latitude = o.getLatitude();
047    Double longitude = o.getLongitude();
048    Batch<DensityTile> batch = new Batch<DensityTile>();
049
050    if (!o.hasSpatialIssue() && MercatorProjectionUtil.isPlottable(latitude, longitude)) {
051      Set<Integer> taxa =
052        Sets.newHashSet(o.getKingdomKey(), o.getPhylumKey(), o.getClassKey(), o.getOrderKey(), o.getFamilyKey(),
053          o.getGenusKey(), o.getSpeciesKey(), o.getTaxonKey());
054
055      // locate the tile
056      int tileX = MercatorProjectionUtil.toTileX(o.getLongitude(), zoom);
057      int tileY = MercatorProjectionUtil.toTileY(o.getLatitude(), zoom);
058
059
060      for (Integer id : taxa) {
061        if (id != null) {
062          addMutations(batch, TileContentType.TAXON, String.valueOf(id), tileX, tileY, zoom, pixelsPerCluster, op, o);
063        }
064      }
065      if (o.getPublishingOrgKey() != null) {
066        addMutations(batch, TileContentType.PUBLISHER, String.valueOf(o.getPublishingOrgKey()), tileX, tileY, zoom,
067          pixelsPerCluster, op, o);
068      }
069      if (o.getDatasetKey() != null) {
070        addMutations(batch, TileContentType.DATASET, String.valueOf(o.getDatasetKey()), tileX, tileY, zoom,
071          pixelsPerCluster, op, o);
072      }
073      if (o.getCountry() != null) {
074        addMutations(batch, TileContentType.COUNTRY, o.getCountry().getIso2LetterCode(), tileX, tileY, zoom,
075          pixelsPerCluster, op, o);
076      }
077      if (o.getPublishingCountry() != null) {
078        addMutations(batch, TileContentType.PUBLISHING_COUNTRY, o.getPublishingCountry().getIso2LetterCode(), tileX, tileY, zoom,
079          pixelsPerCluster, op, o);
080      }
081    }
082    return batch;
083  }
084}
