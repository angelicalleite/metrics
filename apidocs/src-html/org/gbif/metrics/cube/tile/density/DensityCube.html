001package org.gbif.metrics.cube.tile.density;
002
003import org.gbif.metrics.cube.tile.bucketer.MapKeyBucketer;
004import org.gbif.metrics.cube.tile.io.TileContentType;
005
006import java.util.List;
007
008import com.google.common.collect.ImmutableList;
009import com.urbanairship.datacube.DataCube;
010import com.urbanairship.datacube.Dimension;
011import com.urbanairship.datacube.Rollup;
012import com.urbanairship.datacube.bucketers.BigEndianIntBucketer;
013import com.urbanairship.datacube.bucketers.EnumToOrdinalBucketer;
014
015/**
016 * The cube definition for a density map.
017 */
018public class DensityCube {
019
020  public static final Dimension<TileContentType> TYPE = new Dimension<TileContentType>("type", new EnumToOrdinalBucketer<TileContentType>(1), false,
021    1);
022  public static final Dimension<String> KEY = new Dimension<String>("key", new MapKeyBucketer(), false, MapKeyBucketer.BYTES);
023  public static final Dimension<Integer> ZOOM = new Dimension<Integer>("zoom", new BigEndianIntBucketer(), false, 4);
024  public static final Dimension<Integer> TILE_X = new Dimension<Integer>("tileX", new BigEndianIntBucketer(), false, 4);
025  public static final Dimension<Integer> TILE_Y = new Dimension<Integer>("tileY", new BigEndianIntBucketer(), false, 4);
026
027  // Singleton instance if accessed through the instance() method
028  public static final DataCube<DensityTile> INSTANCE = newInstance();
029
030  // Not for instantiation
031  private DensityCube() {
032  }
033
034  private static DataCube<DensityTile> newInstance() {
035    List<Dimension<?>> dimensions = ImmutableList.<Dimension<?>>of(TYPE, KEY, ZOOM, TILE_X, TILE_Y);
036    List<Rollup> rollups = ImmutableList.of(new Rollup(TYPE, KEY, ZOOM, TILE_X, TILE_Y));
037    return new DataCube<DensityTile>(dimensions, rollups);
038  }
039}
