022/**
023 * A density tile clusters counts georeferenced items to a configurable sized cluster of pixels,
024 * and to a specific layer. For example, a density tile might contain data to 4px groupings
025 * for specimens collected this decade, and also observations made within this decade.
026 * For example, one might cluster to 4px by 4px. This class wraps an Avro
027 * object which is used for efficient SerDe and implements the CubeIo operation
028 * interface to allow it to be used for cube additions.
029 */
030public class DensityTile implements Op {
031
032  /**
033   * Simple builder to help produce DensityTiles from lat lng data.
034   * Requires the
035   */
036  public static class Builder {
037
038    private final int x, y, zoom, clusterSize;
039    private final Map<Layer, Map<Integer, Integer>> layers = Maps.newHashMap();
040
041    public Builder(int zoom, int x, int y, int clusterSize) {
042      this.x = x;
043      this.y = y;
044      this.zoom = zoom;
045      this.clusterSize = clusterSize;
046    }
047
048    public DensityTile build() {
049      return new DensityTile(x,y,zoom,clusterSize, layers);
050    }
051
052    public Builder collect(Layer layer, double lat, double lng, int count) {
053      // ignore zero which means no content
054      if (count != 0) {
055        // assert that we are on the correct tile
056        Point t = MercatorProjectionUtil.toTileXY(lat, lng, zoom);
057        if (t.x == x && t.y == y) {
058          int offsetX = MercatorProjectionUtil.getOffsetX(lat, lng, zoom);
059          int offsetY = MercatorProjectionUtil.getOffsetY(lat, lng, zoom);
060          int id = toCellId(offsetX, offsetY, clusterSize);
061
062          // create the grid if necessary
063          if (!layers.containsKey(layer)) {
064            Map<Integer, Integer> g = Maps.newHashMap();
065            layers.put(layer, g);
066          }
067          Map<Integer, Integer> cells = layers.get(layer);
068
069          if (cells.containsKey(id)) {
070            cells.put(id, cells.get(id) + count);
071          } else {
072            cells.put(id, count);
073          }
074        }
075      }
076      return this;
077    }
078
079    public int getX() {
080      return x;
081    }
082
083    public int getY() {
084      return y;
085    }
086
087    public int getZoom() {
088      return zoom;
089    }
090  }

131  /**
132   * Utility to deserialize an Avro based serialization.
133   *
134   * @see DensityTileAvro
135   */
136  public static DensityTile DESERIALIZE(byte[] b) throws IOException {
137    DensityTileAvro avro = DensityTileAvroSerDeUtils.decode(b);
138    Map<CharSequence, GridAvro> m = avro.getGrids();
139    DensityTile dt = new DensityTile(avro.getX(), avro.getY(), avro.getZoom(), avro.getClusterSize());
140
141    // recreate the grid structure
142    for (Entry<CharSequence, GridAvro> e : m.entrySet()) {
143      Map<Integer, Integer> cells = Maps.newHashMap();
144      dt.layers.put(Layer.valueOf(e.getKey().toString()), cells);
145      // populate the cells within the grid just created
146      for (Entry<CharSequence, Integer> c : e.getValue().getCells().entrySet()) {
147        cells.put(Integer.parseInt(c.getKey().toString()), c.getValue());
148      }
149    }
150    return dt;
151  }

153  /**
154   * Based on the cluster size, generates the cell id from the offset X and Y within
155   * the tile. This is a linear form of the usual google tile addressing scheme, where
156   * the top row goes left to right 0,1,2,3 etc for as many cells as are in the row.
157   */
158  public static int toCellId(int x, int y, int clusterSize) {
159    int tpc = TILE_SIZE / clusterSize;
160    return (x / clusterSize) + (tpc * (y / clusterSize));
161  }

163  @Override
164  public Op add(Op otherOp) {
165    if (otherOp instanceof DensityTile) {
166      DensityTile o = (DensityTile) otherOp;
167
168      if (o.clusterSize != clusterSize) {
169        throw new IllegalArgumentException("Cannot merge DensityTiles with different cluster sizes. Supplied "
170          + o.clusterSize + " and " + clusterSize);
171      }
172
173      // build a new map merging the densities from the provided cells
174      Map<Layer, Map<Integer, Integer>> layers = Maps.newHashMap(layers());
175      for (Entry<Layer, Map<Integer, Integer>> g : o.layers.entrySet()) {
176        // If we don't have it already, copy it in, otherwise merge it in
177        if (!layers.containsKey(g.getKey())) {
178          layers.put(g.getKey(), g.getValue());
179        } else {
180          Map<Integer, Integer> cells = layers.get(g.getKey());
181          for (Entry<Integer, Integer> c : g.getValue().entrySet()) {
182            // merge in the cell counts
183            if (cells.containsKey(c.getKey())) {
184              int count = cells.get(c.getKey()) + c.getValue();
185              // as data is removed, it is feasible that we end up with no content - save space and remove 0 counts
186              if (count == 0) {
187                cells.remove(c.getKey());
188              } else {
189                cells.put(c.getKey(), count);
190              }
191
192            } else {
193              cells.put(c.getKey(), c.getValue());
194            }
195          }
196        }
197      }
198
199      return new DensityTile(x,y,zoom,clusterSize, layers);
200
201    } else {
202      throw new IllegalArgumentException(
203        "Cannot merge DensityTiles when supplied tile is not a density tile.  Supplied: " + otherOp.getClass());
204    }
205  }

207  /**
208   * Looks up a cell value using the encoded form, returning 0 for anything not found.
209   */
210  public int cell(Layer grid, int cellId) {
211    Map<Integer, Integer> cells = layers.get(grid);
212    if (cells != null) {
213      Integer v = cells.get(cellId);
214      return (v == null) ? 0 : v;
215    }
216    return 0;
217  }

220  /**
221   * Looks up a cell value, returning 0 for anything not found.
222   */
223  public int cell(Layer grid, int x, int y) {
224    int cellId = x + (TILE_SIZE / clusterSize * y);
225    return cell(grid, cellId);
226  }

228  /**
229   * A utility to downscale from e.g. 1px resolution to 2px or 4px resolutions.
230   * @return a new DensityTile or this, if the there is no work to do
231   */
232  public DensityTile downscale(int newClusterSize) {
233    Preconditions.checkArgument(SUPPORTED_RESOLUTIONS.contains(newClusterSize),
234      "Supplied resolution [%s] is not supported.  Supported resolutions %s", newClusterSize, SUPPORTED_RESOLUTIONS);
235    Preconditions.checkArgument(newClusterSize>=clusterSize,
236      "Can only downscale resolutions.  Current resolution[%s] is coarser than that desired[%s]", clusterSize, newClusterSize);
237
238    if (getClusterSize() == newClusterSize) {
239      return this; // nothing to do
240    } else {
241      // to downscale we basically work out the x,y in the current grid system and then project that onto the new one,
242      // by working out how many bits difference there are
243      int bitShift = 0;
244      for (int i=newClusterSize; i>clusterSize; i = i>>1, bitShift++) {
245        ;
246      }
247      int cellsPerRow = TILE_SIZE / clusterSize;
248      int newCellsPerRow = TILE_SIZE / newClusterSize;
249
250      // build a new map scaling all the densities from the provided cells
251      Map<Layer, Map<Integer, Integer>> newLayers = Maps.newHashMap();
252      for (Entry<Layer, Map<Integer, Integer>> g : layers.entrySet()) {
253
254        Map<Integer, Integer> cells = newLayers.get(g.getKey());
255
256        // If we don't have it already, create it
257        if (cells == null) {
258          cells = Maps.newHashMap();
259          newLayers.put(g.getKey(), cells);
260        }
261
262        // downscale and merge in
263        for (Entry<Integer, Integer> c : g.getValue().entrySet()) {
264          // downscale here (written verbosely to aid future versions of me)
265          int currentKey = c.getKey();
266          int currentX = currentKey % cellsPerRow;
267          int currentY = currentKey / cellsPerRow;
268          int newX = currentX>>bitShift;
269          int newY = currentY>>bitShift;
270          int newKey = (newY * newCellsPerRow) + newX;
271
272          // merge in the cell counts
273          if (cells.containsKey(newKey)) {
274            int count = cells.get(newKey) + c.getValue();
275            cells.put(newKey, count);
276          } else {
277            cells.put(newKey, c.getValue());
278          }
279        }
280      }
281
282      return new DensityTile(x,y,zoom, newClusterSize, newLayers);
283
284    }
285  }

295  /**
296   * Serializes using an Avro format.
297   *
298   * @see DensityTileAvro
299   */
300  @Override
301  public byte[] serialize() {
302    DensityTileAvro avro = new DensityTileAvro();
303    avro.setClusterSize(clusterSize);
304    Map<CharSequence, GridAvro> g = Maps.newHashMap();
305    avro.setGrids(g);
306    // Avro only supports String typed maps, hence these lines of workaround type casting
307    for (Entry<Layer, Map<Integer, Integer>> layer : layers.entrySet()) {
308      Map<CharSequence, Integer> m = Maps.newHashMap();
309      for (Entry<Integer, Integer> e : layer.getValue().entrySet()) {
310        m.put(String.valueOf(e.getKey()), e.getValue());
311      }
312      g.put(layer.getKey().toString(), GridAvro.newBuilder().setCells(m).build());
313    }
314
315    try {
316      return DensityTileAvroSerDeUtils.encode(avro);
317    } catch (IOException e) {
318      // We can't throw a checked exception, so repackage
319      throw new RuntimeException("Unable to serialize: " + e);
320    }
321  }
