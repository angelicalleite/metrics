001package org.gbif.metrics.cube.tile.density.backfill;
002
003import org.gbif.metrics.cube.HBaseCubes;
004import org.gbif.metrics.cube.HBaseSourcedBackfill;
005import org.gbif.metrics.cube.mapred.OccurrenceWritable;
006import org.gbif.metrics.cube.tile.density.DensityCube;
007import org.gbif.metrics.cube.tile.density.DensityTile;
008import org.gbif.metrics.cube.tile.density.Layer;
009import org.gbif.metrics.cube.tile.io.TileKeyWritable;
010
011import java.io.IOException;
012
013import com.urbanairship.datacube.WriteBuilder;
014import org.apache.hadoop.conf.Configuration;
015import org.apache.hadoop.hbase.util.Bytes;
016import org.apache.hadoop.io.NullWritable;
017import org.apache.hadoop.mapreduce.Reducer;
018
019/**
020 * Collects the tile, and writes to the cube.
021 */
022public class CubeWriterReducer extends Reducer<TileKeyWritable, OccurrenceWritable, NullWritable, NullWritable> {
023
024  private int pixelsPerCluster = Backfill.DEFAULT_PIXELS_PER_CLUSTER;
025  private HBaseCubes<DensityTile> hbaseCube;
026
027  @Override
028  protected void cleanup(Context context) throws IOException, InterruptedException {
029    super.cleanup(context);
030    hbaseCube.close();
031  }
032
033  /**
034   * Builds the tile, and pushes to the cube.
035   */
036  @Override
037  protected void reduce(TileKeyWritable k, Iterable<OccurrenceWritable> occurrences, Context context)
038    throws IOException {
039    DensityTile.Builder b = DensityTile.builder(k.getZ(), k.getX(), k.getY(), pixelsPerCluster);
040    int count = 0;
041    for (OccurrenceWritable o : occurrences) {
042      b.collect(Layer.inferFrom(o), o.getLatitude(), o.getLongitude(), o.getCount());
043      count++;
044    }
045    DensityTile t = b.build();
046
047    context.setStatus("Type[" + k.getType() + "],  Key[" + k.getKey() + "], Z[" + k.getZ() + "], X[" + k.getX()
048      + "], Y[" + k.getY() + "], Count[" + count + "]");
049
050    try {
051
052      WriteBuilder wb =
053        new WriteBuilder(DensityCube.INSTANCE).at(DensityCube.ZOOM, k.getZ()).at(DensityCube.TILE_X, k.getX())
054          .at(DensityCube.TILE_Y, k.getY()).at(DensityCube.KEY, k.getKey()).at(DensityCube.TYPE, k.getType());
055      hbaseCube.write(t, wb);
056
057    } catch (Exception e) {
058      throw new IOException(e);
059    }
060  }
061
062  @Override
063  protected void setup(Context context) throws IOException, InterruptedException {
064    super.setup(context);
065    Configuration conf = context.getConfiguration();
066    pixelsPerCluster = conf.getInt(Backfill.KEY_PIXELS_PER_CLUSTER, Backfill.DEFAULT_PIXELS_PER_CLUSTER);
067    int writeBatchSize = conf.getInt(Backfill.KEY_WRITE_BATCH_SIZE, Backfill.DEFAULT_WRITE_BATCH_SIZE);
068    hbaseCube =
069      HBaseCubes.newOverwritingBatchAsync(DensityCube.INSTANCE, DensityTile.DESERIALIZER,
070        Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_BACKFILL_TABLE)),
071        Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_CF)), conf, writeBatchSize);
072  }
073}
