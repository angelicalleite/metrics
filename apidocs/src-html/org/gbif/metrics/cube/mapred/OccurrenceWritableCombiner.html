001package org.gbif.metrics.cube.mapred;
002
003import java.io.IOException;
004
005import org.apache.hadoop.io.IntWritable;
006import org.apache.hadoop.mapreduce.Reducer;
007
008/**
009 * Accumulates the counts of occurrences, but DOES NOT set the count on the occurrence itself.
010 * This is required so that it can have several passes, and the keys remain the same.
011 */
012public class OccurrenceWritableCombiner extends Reducer<OccurrenceWritable, IntWritable, OccurrenceWritable, IntWritable> {
013
014  private final IntWritable i = new IntWritable();
015
016  @Override
017  protected void reduce(OccurrenceWritable key, Iterable<IntWritable> value, Context context) throws IOException,
018    InterruptedException {
019    int total = 0;
020    for (IntWritable i : value) {
021      total += i.get();
022    }
023    context.setStatus("Occurrence from dataset[" + key.getDatasetKey() + "],  Taxon[" + key.getTaxonID() + "], Count["
024      + total + "]");
025    i.set(total);
026    context.write(key, i);
027  }
028}
