001package org.gbif.metrics.cube.occurrence;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.api.vocabulary.EndpointType;
006import org.gbif.metrics.cube.bucketer.BasisOfRecordBucketer;
007import org.gbif.metrics.cube.bucketer.CountryBucketer;
008import org.gbif.metrics.cube.bucketer.EndpointTypeBucketer;
009import org.gbif.metrics.cube.bucketer.UUIDBucketer;
010
011import java.util.List;
012import java.util.Map;
013import java.util.UUID;
014
015import com.google.common.collect.ImmutableList;
016import com.google.common.collect.ImmutableList.Builder;
017import com.google.common.collect.ImmutableMap;
018import com.google.common.collect.ImmutableSet;
019import com.urbanairship.datacube.DataCube;
020import com.urbanairship.datacube.Dimension;
021import com.urbanairship.datacube.DimensionAndBucketType;
022import com.urbanairship.datacube.Rollup;
023import com.urbanairship.datacube.bucketers.BigEndianIntBucketer;
024import com.urbanairship.datacube.bucketers.BooleanBucketer;
025import com.urbanairship.datacube.ops.LongOp;

027/**
028 * The INTERNAL data cube definition for occurrence records.  
029 * This is the low level representation and changes should be carefully considered in advance considering the following: 
030 * <ol>
031 * <li>All public API Dimensions should be declared here and with the correct type</li>
032 * <li>The API mapping must be complete</li>
033 * </ol>
034 * 
035 * Note: It was by design to expose different objects in the public API, as the datacube objects are too complex, and 
036 * datacube brings in too many dependencies.  This also included the acceptance of collision in class names, but this is 
037 * constrained to a 2 places (here, and the HTTP parameter mapping in the web services) which was deemed acceptable for a nice
038 * public API. 
039 */
040public class OccurrenceCube {
041  // number of bytes needed to store the types 
042  private final static int INT_BYTES = 4;
043  private final static int BOOLEAN_BYTES = 1;
044  
045  // NOTE: Ensure that these are in sync with the public API, including the correct types
046  // No ID substitution in place, and all dimensions are optional 
047  public static final Dimension<Country> COUNTRY = new Dimension<Country>("country", new CountryBucketer(), false, CountryBucketer.BYTES, true);
048  public static final Dimension<Integer> YEAR = new Dimension<Integer>("year", new BigEndianIntBucketer(), false, INT_BYTES, true);
049  public static final Dimension<Boolean> IS_GEOREFERENCED = new Dimension<Boolean>("georeferenced", new BooleanBucketer(), false, BOOLEAN_BYTES, true);
050  public static final Dimension<BasisOfRecord> BASIS_OF_RECORD = new Dimension<BasisOfRecord>("basisOfRecord", new BasisOfRecordBucketer(), false, BasisOfRecordBucketer.BYTES, true);
051  public static final Dimension<Country> HOST_COUNTRY = new Dimension<Country>("hostCountry", new CountryBucketer(), false, CountryBucketer.BYTES, true);
052  public static final Dimension<UUID> DATASET_KEY = new Dimension<UUID>("datasetKey", new UUIDBucketer(), false, UUIDBucketer.BYTES, true);  
053  public static final Dimension<Integer> NUB_KEY = new Dimension<Integer>("nubKey", new BigEndianIntBucketer(), false, INT_BYTES, true);  
054  public static final Dimension<EndpointType> PROTOCOL = new Dimension<EndpointType>("protocol", new EndpointTypeBucketer(), false, EndpointTypeBucketer.BYTES,true);
055
056  // Index mapping the public API cube dimensions to the internal datacube dimensions
057  // NOTE: Ensure ALL public API mappings are covered
058  public static final Map<org.gbif.api.model.metrics.cube.Dimension<?>, Dimension<?>> API_MAPPING = 
059    ImmutableMap.<org.gbif.api.model.metrics.cube.Dimension<?>, Dimension<?>>builder()
060      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.IS_GEOREFERENCED, IS_GEOREFERENCED)
061      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.BASIS_OF_RECORD, BASIS_OF_RECORD)
062      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.COUNTRY, COUNTRY)
063      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.HOST_COUNTRY, HOST_COUNTRY)
064      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.DATASET_KEY, DATASET_KEY)      
065      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.NUB_KEY, NUB_KEY)
066      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.PROTOCOL, PROTOCOL)
067      .put(org.gbif.api.model.metrics.cube.OccurrenceCube.YEAR, YEAR)
068    .build();
069    
070  // Singleton instance 
071  public static final DataCube<LongOp> INSTANCE = newInstance();
072
073  // Not for instantiation
074  private OccurrenceCube() {
075  }
076  
077  /**
078   * This uses the CubeIo as defined in the public API to construct an internal DataCube definition.
079   * @return A new instance of the cube
080   */
081  private static DataCube<LongOp> newInstance() {
082    // Dimensions as defined in the public API
083    List<Dimension<?>> dimensions = 
084      ImmutableList.<Dimension<?>>copyOf(API_MAPPING.values()); // defensive copy
085    
086    // Rollups generated as defined in the public API
087    // This could be done manually, but by using the API definition, we safeguard against developer oversight
088    Builder<Rollup> b = ImmutableList.<Rollup>builder();
089    for (org.gbif.api.model.metrics.cube.Rollup r : org.gbif.api.model.metrics.cube.OccurrenceCube.ROLLUPS) {
090      com.google.common.collect.ImmutableSet.Builder<DimensionAndBucketType> sb = ImmutableSet.<DimensionAndBucketType>builder();
091      for (org.gbif.api.model.metrics.cube.Dimension<?> d : r.getDimensions()) {
092        sb.add(new DimensionAndBucketType(API_MAPPING.get(d)));
093      }
094      b.add(new Rollup(sb.build()));
095    }
096    List<Rollup> rollups = b.build();
097
098    return new DataCube<LongOp>(dimensions, rollups);
099  }
100}
