018import static org.gbif.metrics.cube.occurrence.OccurrenceCube.BASIS_OF_RECORD;
019import static org.gbif.metrics.cube.occurrence.OccurrenceCube.COUNTRY;
020import static org.gbif.metrics.cube.occurrence.OccurrenceCube.DATASET_KEY;
021import static org.gbif.metrics.cube.occurrence.OccurrenceCube.HOST_COUNTRY;
022import static org.gbif.metrics.cube.occurrence.OccurrenceCube.IS_GEOREFERENCED;
023import static org.gbif.metrics.cube.occurrence.OccurrenceCube.NUB_KEY;
024import static org.gbif.metrics.cube.occurrence.OccurrenceCube.PROTOCOL;
025import static org.gbif.metrics.cube.occurrence.OccurrenceCube.YEAR;

027/**
028 * A utility class to encapsulate the logic needed to convert between the denormalized
029 * format of occurrences (which have kingdomKey, phylumKey etc) and the normalized format
030 * used by the occurrence cube (which uses only a nubKey).
031 * This class uses the occurrence cube rollup definitions to determine which cube addresses
032 * are to be updated for a given record. This is only necessary due to the normalization
033 * that is required to for the taxonomic ranks that are all merged into a single nub key.
034 * Brief explanation:
035 * - Imagine a record from dataset:1 with kingdom:1 and phylum:2
036 * - Imagine the cube counts "by dataset", "records by nubKey" and "records by dataset by nubKey"
037 * We need to update the following cube addresses:
038 * - "dataset:1"
039 * - "nubKey:1"
040 * - "nubKey:2"
041 * - "dataset:1,nubKey:1"
042 * - "dataset:1,nubKey:2"
043 * You cannot update the cube 2 times for each taxon, or else the dataset count will be 2 in this
044 * example.
045 * This is the problem this class addresses, by providing the addresses for a single occurrence record.
046 */
047public class OccurrenceAddressUtil {
048
049  /**
050   * For the given occurrence, determines the mutations (addresses and operations) that need
051   * to be applied.
052   *
053   * @param occurrence The denormalized representation
054   * @param op That is going to be applied to the cube
055   * @return The batch of updates to apply
056   */
057  public static Batch<LongOp> cubeMutation(Occurrence occurrence, LongOp op) {
058    // Needs mutability, since we rely on overwriting as a lazy distinction mechanism
059    // such as for when speciesKey = nubKey
060    Map<Address, LongOp> m = Maps.newHashMap();
061    m.putAll(mutationsForTaxon(occurrence, occurrence.getKingdomKey(), op));
062    m.putAll(mutationsForTaxon(occurrence, occurrence.getPhylumKey(), op));
063    m.putAll(mutationsForTaxon(occurrence, occurrence.getClassKey(), op));
064    m.putAll(mutationsForTaxon(occurrence, occurrence.getOrderKey(), op));
065    m.putAll(mutationsForTaxon(occurrence, occurrence.getFamilyKey(), op));
066    m.putAll(mutationsForTaxon(occurrence, occurrence.getGenusKey(), op));
067    m.putAll(mutationsForTaxon(occurrence, occurrence.getSpeciesKey(), op));
068    m.putAll(mutationsForTaxon(occurrence, occurrence.getTaxonKey(), op));
069    return new Batch<LongOp>(m);
070  }
071
072  /**
073   * For the given occurrence writable, determines the mutations (addresses and operations) that need
074   * to be applied.
075   *
076   * @param occurrence The writable representation
077   * @param op That is going to be applied to the cube
078   * @return The batch of updates to apply
079   */
080  public static Batch<LongOp> cubeMutation(OccurrenceWritable occurrence, LongOp op) {
081    // Needs mutability, since we rely on overwriting as a lazy distinction mechanism
082    // such as for when speciesKey = nubKey
083    Map<Address, LongOp> m = Maps.newHashMap();
084    m.putAll(mutationsForTaxon(occurrence, occurrence.getKingdomID(), op));
085    m.putAll(mutationsForTaxon(occurrence, occurrence.getPhylumID(), op));
086    m.putAll(mutationsForTaxon(occurrence, occurrence.getClassID(), op));
087    m.putAll(mutationsForTaxon(occurrence, occurrence.getOrderID(), op));
088    m.putAll(mutationsForTaxon(occurrence, occurrence.getFamilyID(), op));
089    m.putAll(mutationsForTaxon(occurrence, occurrence.getGenusID(), op));
090    m.putAll(mutationsForTaxon(occurrence, occurrence.getSpeciesID(), op));
091    m.putAll(mutationsForTaxon(occurrence, occurrence.getTaxonID(), op));
092    return new Batch<LongOp>(m);
093  }
094
095  /**
096   * For the given taxon, returns all the updates to be applied to the cube
097   */
098  private static Map<Address, LongOp> mutationsForTaxon(Occurrence occurrence, Integer nubKey, LongOp op) {
099    WriteBuilder wb = new WriteBuilder(OccurrenceCube.INSTANCE);
100    addEnumDimension(wb, occurrence.getCountry(), COUNTRY);
101    addEnumDimension(wb, occurrence.getProtocol(), PROTOCOL);
102    addGeoreferencingDimension(wb, occurrence, IS_GEOREFERENCED);
103    addEnumDimension(wb, occurrence.getBasisOfRecord(), BASIS_OF_RECORD);
104    addUUIDDimension(wb, occurrence.getDatasetKey(), DATASET_KEY);
105    addEnumDimension(wb, occurrence.getPublishingCountry(), HOST_COUNTRY);
106    addIntDimension(wb, nubKey, NUB_KEY);
107    addIntDimension(wb, occurrence.getYear(), YEAR);
108    return OccurrenceCube.INSTANCE.getWrites(wb, op).getMap();
109  }
110
111  /**
112   * For the given taxon, returns all the updates to be applied to the cube
113   */
114  private static Map<Address, LongOp> mutationsForTaxon(OccurrenceWritable occurrence, Integer nubKey, LongOp op) {
115    WriteBuilder wb = new WriteBuilder(OccurrenceCube.INSTANCE);
116    addCountryEnumDimension(wb, occurrence.getCountryIsoCode(), COUNTRY);
117    addEnumDimension(wb, occurrence.getProtocol(), PROTOCOL);
118    addGeoreferencingDimension(wb, occurrence, IS_GEOREFERENCED);
119    addEnumDimension(wb, occurrence.getBasisOfRecord(), BASIS_OF_RECORD);
120    addUUIDDimension(wb, occurrence.getDatasetKey(), DATASET_KEY);
121    addCountryEnumDimension(wb, occurrence.getHostCountryIsoCode(), HOST_COUNTRY);
122    addIntDimension(wb, nubKey, NUB_KEY);
123    addIntDimension(wb, occurrence.getYear(), YEAR);
124    return OccurrenceCube.INSTANCE.getWrites(wb, op).getMap();
125  }
126
127  private static WriteBuilder addGeoreferencingDimension(WriteBuilder wb, Occurrence occurrence,
128                                                         Dimension<Boolean> isGeoreferenced) {
129    return (occurrence.getLatitude() != null && occurrence.getLongitude() != null && !occurrence.hasSpatialIssue()) ?
130      wb.at(IS_GEOREFERENCED, true) : wb.at(IS_GEOREFERENCED, false);
131  }
132
133  private static WriteBuilder addGeoreferencingDimension(WriteBuilder wb, OccurrenceWritable occurrence,
134                                                         Dimension<Boolean> isGeoreferenced) {
135    return (occurrence.getLatitude() != null && occurrence.getLongitude() != null && !occurrence.hasSpatialIssue()) ?
136      wb.at(IS_GEOREFERENCED, true) : wb.at(IS_GEOREFERENCED, false);
137  }
138
139  private static WriteBuilder addUUIDDimension(WriteBuilder wb, UUID i, Dimension<UUID> dim) {
140    return (i != null) ? wb.at(dim, i) : wb;
141  }
142
143  private static WriteBuilder addUUIDDimension(WriteBuilder wb, String s, Dimension<UUID> dim) {
144    return (s != null) ? wb.at(dim, UUID.fromString(s)) : wb;
145  }
146
147  private static WriteBuilder addIntDimension(WriteBuilder wb, Integer i, Dimension<Integer> dim) {
148    return (i != null) ? wb.at(dim, i) : wb;
149  }
150
151  private static <T extends Enum<?>> WriteBuilder addEnumDimension(WriteBuilder wb, T e, Dimension<T> dim) {
152    return (e != null) ? wb.at(dim, e) : wb;
153  }
154
155  private static <T extends Enum<?>> WriteBuilder addCountryEnumDimension(WriteBuilder wb, String s,
156    Dimension<Country> dim) {
157    if (!Strings.isNullOrEmpty(s)) {
158      Country c = Country.fromIsoCode(s);
159      return (c != null) ? wb.at(dim, c) : wb; // skip unparsable countries
160    }
161    return wb;
162  }
