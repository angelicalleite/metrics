001package org.gbif.metrics.cube.occurrence.backfill;
002
003import org.gbif.metrics.cube.mapred.OccurrenceWritable;
004import org.gbif.metrics.cube.occurrence.OccurrenceAddressUtil;
005
006import java.io.IOException;
007
008import com.urbanairship.datacube.Address;
009import com.urbanairship.datacube.Batch;
010import com.urbanairship.datacube.ops.LongOp;
011import org.apache.commons.lang.ArrayUtils;
012import org.apache.hadoop.hbase.client.Result;
013import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
014import org.apache.hadoop.hbase.mapreduce.TableMapper;
015import org.apache.hadoop.io.IntWritable;
016
017/**
018 * Reads the table, emitting the cube address to be updated and ONE for the count.
019 */
020public class TableReaderMapper extends TableMapper<ImmutableBytesWritable, IntWritable> {
021
022  private final IntWritable ONE = new IntWritable(1);
023  private final ImmutableBytesWritable KEY = new ImmutableBytesWritable();
024  private static final byte[] CUBE_NAME = ArrayUtils.EMPTY_BYTE_ARRAY;
025
026
027  @Override
028  protected void map(ImmutableBytesWritable key, Result row, Context context) throws IOException, InterruptedException {
029    
030    OccurrenceWritable o = OccurrenceWritable.newInstance(row);
031    
032    // determine the address
033    Batch<LongOp> addresses = OccurrenceAddressUtil.cubeMutation(o, new LongOp(1));
034    for (Address a : addresses.getMap().keySet()) {
035      byte[] rowKey = ArrayUtils.addAll(CUBE_NAME, a.toKey(null));
036      KEY.set(rowKey);  
037      context.write(KEY, ONE);
038    }
039    context.setStatus("Occurrence from dataset[" + o.getDatasetKey() + "] produced [" + addresses.getMap().size() + "] cube mutations");    
040  }
041
042  @Override
043  protected void setup(Context context) throws IOException, InterruptedException {
044    super.setup(context);
045  }
046}
