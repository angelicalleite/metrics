001package org.gbif.metrics.cube.occurrence.backfill;
002
003import org.gbif.metrics.cube.HBaseSourcedBackfill;
004
005import java.io.IOException;
006import java.util.concurrent.TimeUnit;
007
008import com.urbanairship.datacube.Util;
009import com.yammer.metrics.reporting.GangliaReporter;
010import org.apache.commons.lang.ArrayUtils;
011import org.apache.hadoop.conf.Configuration;
012import org.apache.hadoop.hbase.client.Put;
013import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
014import org.apache.hadoop.hbase.mapreduce.TableReducer;
015import org.apache.hadoop.hbase.util.Bytes;
016import org.apache.hadoop.io.IntWritable;
017
018/**
019 * Accumulates the counts and creates table PUTs.
020 */
021public class CubeWriterReducer extends TableReducer<ImmutableBytesWritable, IntWritable, ImmutableBytesWritable> {  
022  private byte[] COL_FAMILY;
023  public final static byte[] QUALIFIER = ArrayUtils.EMPTY_BYTE_ARRAY; 
024  
025  @Override
026  protected void reduce(ImmutableBytesWritable key, Iterable<IntWritable> values, final Context context) throws IOException, InterruptedException {
027    int total = 0;
028    for (IntWritable i : values) {
029      total += i.get();
030    }
031    
032    // create the same PUT that data cube would make, and emit it for the reducer to write
033    Put put = new Put(key.get()).add(COL_FAMILY, QUALIFIER, Util.longToBytes((long) total));
034    context.write(key, put);
035    context.getCounter("GBIF", "Cube mutations").increment(1);
036  }
037
038
039  @Override
040  protected void setup(Context context) throws IOException, InterruptedException {
041    super.setup(context);
042    Configuration conf = context.getConfiguration();
043    COL_FAMILY = Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_CF));
044    
045    // TODO: configurify this
046    GangliaReporter.enable(1, TimeUnit.MINUTES, "b5g2.gbif.org", 8649);
047  }
048}
