001package org.gbif.metrics.cube.occurrence.backfill;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.metrics.cube.HBaseCubes;
005import org.gbif.metrics.cube.HBaseSourcedBackfill;
006import org.gbif.metrics.cube.occurrence.OccurrenceAddressUtil;
007import org.gbif.metrics.cube.occurrence.OccurrenceCube;
008import org.gbif.occurrencestore.persistence.util.OccurrenceBuilder;
009
010import java.io.IOException;
011
012import com.google.common.base.Preconditions;
013import com.urbanairship.datacube.Batch;
014import com.urbanairship.datacube.ops.LongOp;
015import org.apache.hadoop.conf.Configuration;
016import org.apache.hadoop.hbase.client.Result;
017import org.apache.hadoop.hbase.io.ImmutableBytesWritable;
018import org.apache.hadoop.hbase.mapreduce.TableMapper;
019import org.apache.hadoop.hbase.util.Bytes;
020import org.apache.hadoop.io.NullWritable;
021
022/**
023 * @deprecated In favor of {@link TableReaderMapper} and {@link CubeWriteReducer}
024 */
025@Deprecated
026public class CubeWriterMapper extends TableMapper<NullWritable, NullWritable> {
027
028  private HBaseCubes<LongOp> cube;
029
030  @Override
031  protected void cleanup(Context context) throws IOException, InterruptedException {
032    super.cleanup(context);
033    cube.close();
034  }
035
036  @Override
037  protected void map(ImmutableBytesWritable key, Result row, Context context) throws IOException, InterruptedException {
038    try {
039      Occurrence occurrence = OccurrenceBuilder.buildOccurrence(row);
040      Preconditions.checkNotNull(occurrence,
041        "Unable to generate an Occurrence from the HBase using the OccurrenceBuilder");
042      Batch<LongOp> update = OccurrenceAddressUtil.cubeMutation(occurrence, new LongOp(1));
043
044      context.setStatus("Handling occurrence[" + occurrence.getKey() + "] produced " + update.getMap().size()
045        + " mutations");
046
047      context.getCounter("GBIF", "Cube mutations").increment(update.getMap().size());
048
049      // Note: the cube is a batch async cube to allow this (see setup())
050      cube.write(update);
051
052    } catch (Exception e) {
053      throw new IOException(e);
054    }
055  }
056
057  @Override
058  protected void setup(Context context) throws IOException, InterruptedException {
059    super.setup(context);
060    Configuration conf = context.getConfiguration();
061    int writeBatchSize =
062      conf.getInt(HBaseSourcedBackfill.KEY_WRITE_BATCH_SIZE, HBaseSourcedBackfill.DEFAULT_WRITE_BATCH_SIZE);
063    // this is the basic incrementing as each record increments the existing count per rollup
064    // NOTE: Very importantly this is in BatchAsync to enable the writing in batches
065    cube =
066      HBaseCubes.newIncrementingBatchAsync(OccurrenceCube.INSTANCE, LongOp.DESERIALIZER,
067        Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_BACKFILL_TABLE)),
068        Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_CF)),
069        conf, writeBatchSize);
070  }
071}
