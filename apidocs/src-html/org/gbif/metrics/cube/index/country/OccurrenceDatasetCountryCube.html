001package org.gbif.metrics.cube.index.country;
002
003import org.gbif.api.vocabulary.Country;
004import org.gbif.metrics.cube.bucketer.CountryBucketer;
005import org.gbif.metrics.cube.index.common.UuidIntMap;
006
007import java.util.List;
008
009import com.google.common.collect.ImmutableList;
010import com.urbanairship.datacube.DataCube;
011import com.urbanairship.datacube.Dimension;
012import com.urbanairship.datacube.Rollup;
013
014/**
015 * The cube definition for holding an index of datasets having occurrences.
016 * This cube allows you to retreive in a single call, the complete listing of datasets
017 * and the counts within the dataset.  
018 * Because DataCube does not support any scanning
019 * operations, it is not possible to treat dataset and host country as separate dimensions, 
020 * as one could not ask the cube "what datasets are there for country X".  One can simply
021 * ask for the complete representation for the host country.  This means mutations are expensive
022 * but in this implementation, necessary.   
023 */
024public class OccurrenceDatasetCountryCube {
025  public static final Dimension<Country> COUNTRY = new Dimension<Country>("country", new CountryBucketer(), false, CountryBucketer.BYTES, true);
026
027  // Singleton instance 
028  public static final DataCube<UuidIntMap> INSTANCE = newInstance();
029
030  // Not for instantiation
031  private OccurrenceDatasetCountryCube() {
032  }
033  
034  private static DataCube<UuidIntMap> newInstance() {
035    List<Dimension<?>> dimensions = ImmutableList.<Dimension<?>>of(COUNTRY); 
036    List<Rollup> rollups = ImmutableList.of(new Rollup(COUNTRY));
037    return new DataCube<UuidIntMap>(dimensions, rollups);
038  }
039}
