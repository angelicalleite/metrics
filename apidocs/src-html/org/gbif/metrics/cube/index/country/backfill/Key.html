001package org.gbif.metrics.cube.index.country.backfill;
002
003import org.gbif.api.vocabulary.Country;
004
005import java.io.DataInput;
006import java.io.DataOutput;
007import java.io.IOException;
008import java.util.UUID;
009
010import com.google.common.base.Objects;
011import com.google.common.base.Preconditions;
012import com.google.common.collect.ComparisonChain;
013import org.apache.hadoop.io.WritableComparable;
014
015/**
016 * Synthetic key to use as an intermediary between M and R.
017 */
018public class Key implements WritableComparable<Key> {
019
020  private UUID datasetKey;
021  private Country country;
022
023  public Key() {
024  }
025
026  public Key(UUID datasetKey, Country country) {
027    Preconditions.checkNotNull(datasetKey, "DatasetKey cannot be null");
028    this.datasetKey = datasetKey;
029    this.country = country;
030  }
031
032  @Override
033  public int compareTo(Key o) {
034    return ComparisonChain.start()
035      .compare(country, o.country)
036      .compare(datasetKey, o.datasetKey)
037      .result();
038  }
039
040  @Override
041  public boolean equals(Object object) {
042    if (object instanceof Key) {
043      Key that = (Key) object;
044      return Objects.equal(this.datasetKey, that.datasetKey)
045        && Objects.equal(this.country, that.country);
046    }
047    return false;
048  }
049
050  public Country getCountry() {
051    return country;
052  }
053
054  public UUID getDatasetKey() {
055    return datasetKey;
056  }
057
058  @Override
059  public int hashCode() {
060    return Objects.hashCode(datasetKey, country);
061  }
062
063  @Override
064  public void readFields(DataInput in) throws IOException {
065    country = Country.fromIsoCode(in.readUTF());
066    datasetKey = UUID.fromString(in.readUTF());
067  }
068
069  public void setDatasetKey(UUID datasetKey) {
070    this.datasetKey = datasetKey;
071  }
072
073  public void setNubKey(Country country) {
074    this.country = country;
075  }
076
077  @Override
078  public String toString() {
079    return Objects.toStringHelper(this)
080      .add("datasetKey", datasetKey)
081      .add("country", country)
082      .toString();
083  }
084
085  @Override
086  public void write(DataOutput out) throws IOException {
087    out.writeUTF(country.getIso2LetterCode());
088    out.writeUTF(datasetKey.toString());
089  }
090}
