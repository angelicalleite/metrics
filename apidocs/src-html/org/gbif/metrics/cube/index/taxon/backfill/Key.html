001package org.gbif.metrics.cube.index.taxon.backfill;
002
003import java.io.DataInput;
004import java.io.DataOutput;
005import java.io.IOException;
006import java.util.UUID;
007
008import com.google.common.base.Objects;
009import com.google.common.base.Preconditions;
010import com.google.common.collect.ComparisonChain;
011import org.apache.hadoop.io.WritableComparable;
012
013/**
014 * Synthetic key to use as an intermediary between M and R.
015 */
016public class Key implements WritableComparable<Key> {
017
018  private UUID datasetKey;
019  private int nubKey;
020
021  public Key() {
022  }
023
024  public Key(UUID datasetKey, int nubKey) {
025    Preconditions.checkNotNull(datasetKey, "DatasetKey cannot be null");
026    this.datasetKey = datasetKey;
027    this.nubKey = nubKey;
028  }
029
030  @Override
031  public int compareTo(Key o) {
032    return ComparisonChain.start()
033      .compare(nubKey, o.nubKey)
034      .compare(datasetKey, o.datasetKey)
035      .result();
036  }
037
038  @Override
039  public boolean equals(Object object) {
040    if (object instanceof Key) {
041      Key that = (Key) object;
042      return Objects.equal(this.datasetKey, that.datasetKey)
043        && Objects.equal(this.nubKey, that.nubKey);
044    }
045    return false;
046  }
047
048  public UUID getDatasetKey() {
049    return datasetKey;
050  }
051
052  public int getNubKey() {
053    return nubKey;
054  }
055
056  @Override
057  public int hashCode() {
058    return Objects.hashCode(datasetKey, nubKey);
059  }
060
061  @Override
062  public void readFields(DataInput in) throws IOException {
063    nubKey = in.readInt();
064    datasetKey = UUID.fromString(in.readUTF());
065  }
066
067  public void setDatasetKey(UUID datasetKey) {
068    this.datasetKey = datasetKey;
069  }
070
071  public void setNubKey(int nubKey) {
072    this.nubKey = nubKey;
073  }
074
075  @Override
076  public String toString() {
077    return Objects.toStringHelper(this)
078      .add("datasetKey", datasetKey)
079      .add("nubKey", nubKey)
080      .toString();
081  }
082
083  @Override
084  public void write(DataOutput out) throws IOException {
085    out.writeInt(nubKey);
086    out.writeUTF(datasetKey.toString());
087  }
088}
