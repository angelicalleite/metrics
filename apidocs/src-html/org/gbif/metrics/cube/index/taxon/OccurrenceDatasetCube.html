001package org.gbif.metrics.cube.index.taxon;
002
003import org.gbif.metrics.cube.index.common.UuidIntMap;
004
005import java.util.List;
006
007import com.google.common.collect.ImmutableList;
008import com.urbanairship.datacube.DataCube;
009import com.urbanairship.datacube.Dimension;
010import com.urbanairship.datacube.Rollup;
011import com.urbanairship.datacube.bucketers.BigEndianIntBucketer;
012
013/**
014 * The cube definition for holding an index of datasets having occurrences.
015 * This cube allows you to retreive in a single call, the complete listing of datasets
016 * and the counts within the dataset.  
017 * Because DataCube does not support any scanning
018 * operations, it is not possible to treat dataset and nub key as separate dimensions, 
019 * as one could not ask the cube "what datasets are there for nubKey X".  One can simply
020 * ask for the complete representation for the nub key.  This means mutations are expensive
021 * but in this implementation, necessary.   
022 */
023public class OccurrenceDatasetCube {
024  // number of bytes needed to store the types 
025  private final static int INT_BYTES = 4;
026  public static final Dimension<Integer> NUB_KEY = new Dimension<Integer>("nubKey", new BigEndianIntBucketer(), false, INT_BYTES, true);
027
028  // Singleton instance 
029  public static final DataCube<UuidIntMap> INSTANCE = newInstance();
030
031  // Not for instantiation
032  private OccurrenceDatasetCube() {
033  }
034  
035  private static DataCube<UuidIntMap> newInstance() {
036    List<Dimension<?>> dimensions = ImmutableList.<Dimension<?>>of(NUB_KEY); 
037    List<Rollup> rollups = ImmutableList.of(new Rollup(NUB_KEY));
038    return new DataCube<UuidIntMap>(dimensions, rollups);
039  }
040}
