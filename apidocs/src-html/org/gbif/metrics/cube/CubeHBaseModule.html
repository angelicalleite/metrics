001package org.gbif.metrics.cube;
002
003
004import org.gbif.utils.file.properties.PropertiesUtil;
005
006import java.util.Properties;
007
008import com.google.common.base.Joiner;
009import com.google.inject.AbstractModule;
010
011/**
012 * Utility for reading necessary properties for the guice modules.
013 */
014public abstract class CubeHBaseModule extends AbstractModule {
015
016  private final byte[] cubeTable;
017  private final byte[] counterTable;
018  private final byte[] lookupTable;
019  private final byte[] cf;
020  private final int writeBatchSize;
021
022  public CubeHBaseModule(Properties props, String prefix) {
023    Joiner j = Joiner.on(".");
024    cubeTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, HBaseSourcedBackfill.KEY_CUBE_TABLE), true, null);
025    counterTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, HBaseSourcedBackfill.KEY_COUNTER_TABLE), false, null); // optional
026    lookupTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, HBaseSourcedBackfill.KEY_LOOKUP_TABLE), false, null);  // optional
027    cf = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, HBaseSourcedBackfill.KEY_CF), true, null);
028    writeBatchSize = PropertiesUtil.propertyAsInt(props, j.join(prefix, HBaseSourcedBackfill.KEY_WRITE_BATCH_SIZE), true, null);
029  }
030
031  protected byte[] getCf() {
032    return cf;
033  }
034
035  protected byte[] getCounterTable() {
036    return counterTable;
037  }
038
039  protected byte[] getCubeTable() {
040    return cubeTable;
041  }
042
043  protected byte[] getLookupTable() {
044    return lookupTable;
045  }
046
047  protected int getWriteBatchSize() {
048    return writeBatchSize;
049  }
050}
