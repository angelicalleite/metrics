001package org.gbif.metrics.cube;
002
003import org.gbif.utils.file.properties.PropertiesUtil;
004
005import java.io.IOException;
006import java.util.Properties;
007
008import com.google.common.base.Joiner;
009import com.urbanairship.datacube.Deserializer;
010import com.urbanairship.datacube.backfill.HBaseBackfill;
011import com.urbanairship.datacube.backfill.HBaseBackfillCallback;
012import org.apache.hadoop.conf.Configuration;
013import org.apache.hadoop.hbase.HBaseConfiguration;
014import org.apache.hadoop.hbase.HColumnDescriptor;
015import org.apache.hadoop.hbase.HTableDescriptor;
016import org.apache.hadoop.hbase.client.HBaseAdmin;
017import org.apache.hadoop.hbase.regionserver.StoreFile.BloomType;
018import org.apache.hadoop.hbase.util.Bytes;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * A utility to help write backfillers that scan an HBase table as the input.
024 */
025public class HBaseSourcedBackfill {
026
027  private static final Logger LOG = LoggerFactory.getLogger(HBaseSourcedBackfill.class);
028  private static final String APPLICATION_PROPERTIES = "cube.properties";
029
030  // Configuration that is read from the APPLICATION_PROPERTIES
031  // We use bytes as they are most commonly used but some APIs require
032  // them to be converted back to Strings. This is out of our hands though.
033  private final byte[] cubeTable;
034  private final byte[] snapshotTable;
035  private final byte[] backfillTable;
036  private final byte[] counterTable;
037  private final byte[] lookupTable;
038  private final byte[] cf;
039  private final byte[] sourceTable;
040
041  private final int scannerCache;
042  private final int numReducers;
043  private final int writeBatchSize;
044
045  public static final int DEFAULT_NUM_REDUCERS = 12;
046  public static final int DEFAULT_WRITE_BATCH_SIZE = 1000;
047  public static final int DEFAULT_SCANNER_CACHE = 200;
048
049  // Keys used for the application properties, and in the Hadoop context,
050  // since that is the only way to pass things to the launched MR tasks.
051  public static final String KEY_CUBE_TABLE = "cubeTable";
052  public static final String KEY_SNAPSHOT_TABLE = "snapshotTable";
053  public static final String KEY_BACKFILL_TABLE = "backfillTable";
054  public static final String KEY_COUNTER_TABLE = "counterTable";
055  public static final String KEY_LOOKUP_TABLE = "lookupTable";
056  public static final String KEY_CF = "columnFamily";
057  public static final String KEY_SOURCE_TABLE = "backfillSourceTable";
058  public static final String KEY_SCANNER_CACHE = "backfillScannerCaching";
059  public static final String KEY_HBASE_SCANNER_CACHE = "hbase.client.scanner.caching"; // for HBase job conf
060  public static final String KEY_NUM_REDUCERS = "backfillNumReduceTasks";
061  public static final String KEY_WRITE_BATCH_SIZE = "writeBatchSize";
062
063  private final Properties props;
064
065
066  public HBaseSourcedBackfill(String prefix) throws IllegalArgumentException, IOException {
067    props = PropertiesUtil.loadProperties(APPLICATION_PROPERTIES);
068    Joiner j = null;
069    // handle complete or missing prefixes with no separator
070    if (prefix == null || prefix.endsWith(".")) {
071      j = Joiner.on("").skipNulls();
072    } else {
073      j = Joiner.on('.');
074    }
075
076    cubeTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_CUBE_TABLE), true, null);
077    snapshotTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_SNAPSHOT_TABLE), true, null);
078    backfillTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_BACKFILL_TABLE), true, null);
079    counterTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_COUNTER_TABLE), false, null); // optional
080    lookupTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_LOOKUP_TABLE), false, null); // optional
081    cf = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_CF), true, null);
082    sourceTable = PropertiesUtil.propertyAsUTF8Bytes(props, j.join(prefix, KEY_SOURCE_TABLE), true, null);
083    scannerCache = PropertiesUtil.propertyAsInt(props, j.join(prefix, KEY_SCANNER_CACHE), false, HBaseSourcedBackfill.DEFAULT_SCANNER_CACHE);
084    numReducers = PropertiesUtil.propertyAsInt(props, j.join(prefix, KEY_NUM_REDUCERS), false, DEFAULT_NUM_REDUCERS);
085    writeBatchSize = PropertiesUtil.propertyAsInt(props, j.join(prefix, KEY_WRITE_BATCH_SIZE), false, DEFAULT_WRITE_BATCH_SIZE);
086  }
087
088  /**
089   * Runs the backfill process.
090   * 
091   * @throws IOException On any HBase communication errors
092   */
093  public void backfill(HBaseBackfillCallback callback, Class<? extends Deserializer<?>> deserializer) {
094    Configuration conf = HBaseConfiguration.create();
095    try {
096      setup(conf);
097      HBaseBackfill backfill = new HBaseBackfill(conf, callback, cubeTable, snapshotTable, backfillTable, cf, deserializer);
098
099      backfill.runWithCheckedExceptions();
100      cleanup(conf);
101    } catch (IOException e) {
102      LOG.error("Error running cube backfill", e);
103    }
104  }
105
106  /**
107   * Removes the existing snapshot and backfill tables if present.
108   */
109  private void cleanup(Configuration conf) throws IOException {
110    cleanup(new HBaseAdmin(conf));
111  }
112
113  /**
114   * Removes the existing snapshot and backfill tables if present.
115   */
116  private void cleanup(HBaseAdmin admin) throws IOException {
117    if (admin.tableExists(snapshotTable)) {
118      LOG.info("Deleting table {}", Bytes.toString(snapshotTable));
119      admin.disableTable(snapshotTable);
120      admin.deleteTable(snapshotTable);
121    }
122    if (admin.tableExists(backfillTable)) {
123      LOG.info("Deleting table {}", Bytes.toString(backfillTable));
124      admin.disableTable(backfillTable);
125      admin.deleteTable(backfillTable);
126    }
127  }
128
129  // utility to create a table if it does not exist
130  private void createIfMissing(HBaseAdmin admin, byte[] t) throws IOException {
131    createIfMissing(admin, t, null, null, 0);
132  }
133
134  // utility to create a table if it does not exist
135  private void createIfMissing(HBaseAdmin admin, byte[] t, byte[] startKey, byte[] endKey, int numRegions) throws IOException {
136    if (!admin.tableExists(t)) {
137      LOG.info("Creating table {}", Bytes.toString(t));
138      HColumnDescriptor cfDesc = new HColumnDescriptor(cf);
139      cfDesc.setBloomFilterType(BloomType.NONE);
140      cfDesc.setMaxVersions(1);
141      // TODO: http://dev.gbif.org/issues/browse/MET-7
142      // cfDesc.setCompressionType(Algorithm.SNAPPY); // fails on the Snapshotter at the end
143      HTableDescriptor tableDesc = new HTableDescriptor(t);
144      tableDesc.addFamily(cfDesc);
145      if (startKey != null && endKey != null && numRegions > 0) {
146        admin.createTable(tableDesc, startKey, endKey, numRegions);
147      } else {
148        admin.createTable(tableDesc);
149      }
150    }
151  }
152
153  protected Properties getProperties() {
154    return props;
155  }
156
157  /**
158   * Removes the existing snapshot and backfill tables if present.
159   * Creates the cube,counter and lookup tables if missing.
160   * Sets the configuration options in the Hadoop context.
161   */
162  protected void setup(Configuration conf) throws IOException {
163    HBaseAdmin admin = new HBaseAdmin(conf);
164    cleanup(admin);
165    createIfMissing(admin, cubeTable);
166
167    if (counterTable != null) {
168      createIfMissing(admin, counterTable);
169      conf.set(KEY_COUNTER_TABLE, Bytes.toString(counterTable));  
170    }
171    if (lookupTable != null) {
172      createIfMissing(admin, lookupTable);
173      conf.set(KEY_LOOKUP_TABLE, Bytes.toString(lookupTable));  
174    }
175
176    // unfortunately we need to use Strings again (Hadoop API)
177    conf.set(KEY_CUBE_TABLE, Bytes.toString(cubeTable));
178    conf.set(KEY_SNAPSHOT_TABLE, Bytes.toString(snapshotTable));
179    conf.set(KEY_BACKFILL_TABLE, Bytes.toString(backfillTable));
180    conf.set(KEY_CF, Bytes.toString(cf));
181    conf.set(KEY_SOURCE_TABLE, Bytes.toString(sourceTable));
182    conf.setInt(KEY_HBASE_SCANNER_CACHE, scannerCache);
183    conf.setInt(KEY_NUM_REDUCERS, numReducers);
184    conf.setInt(KEY_WRITE_BATCH_SIZE, writeBatchSize);
185  }
186}
