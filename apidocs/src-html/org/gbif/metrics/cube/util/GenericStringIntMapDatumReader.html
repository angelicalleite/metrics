001package org.gbif.metrics.cube.util;
002
003import java.io.IOException;
004import java.util.Map;
005
006import org.apache.avro.Schema;
007import org.apache.avro.generic.GenericDatumReader;
008import org.apache.avro.io.Decoder;
009
010
011/**
012 * Read data from a Avro Schema.
013 * By defaults deserialized a Charsequence/String object into a String object.
014 * The superclass GenericDatumReader deserializes Strings into Utf8 classes.
015 */
016public class GenericStringIntMapDatumReader extends GenericDatumReader<Map<CharSequence, Integer>> {
017
018  public GenericStringIntMapDatumReader(Schema schema) {
019    super(schema);
020  }
021
022  /**
023   * Overrides the method GenericDatumReader.readString to produce a String instance instead of Utf8 instance.
024   */
025  @Override
026  protected Object readString(Object old, Decoder in) throws IOException {
027    return super.readString(old, in).toString();
028  }
029}
