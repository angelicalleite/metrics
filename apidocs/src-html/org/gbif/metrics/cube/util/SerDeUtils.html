001package org.gbif.metrics.cube.util;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005
006import org.apache.avro.generic.GenericDatumWriter;
007import org.apache.avro.io.BinaryEncoder;
008import org.apache.avro.io.Decoder;
009import org.apache.avro.io.DecoderFactory;
010import org.apache.avro.io.EncoderFactory;
011import org.apache.avro.specific.SpecificDatumReader;
012
013public abstract class SerDeUtils {
014
015  private static final DecoderFactory DECODER_FACTORY = DecoderFactory.get();
016  private static final EncoderFactory ENCODER_FACTORY = EncoderFactory.get();
017
018  protected static <T> T decodeObject(final T object, final byte[] data, final SpecificDatumReader<T> reader) throws IOException {
019    Decoder decoder = DECODER_FACTORY.binaryDecoder(data, null);
020    return reader.read(object, decoder);
021  }
022
023  protected static <T> byte[] encodeObject(final T datum, final GenericDatumWriter<T> writer) throws IOException {
024    // The encoder instantiation can be replaced with a ThreadLocal if needed
025    ByteArrayOutputStream os = new ByteArrayOutputStream();
026    BinaryEncoder encoder = ENCODER_FACTORY.binaryEncoder(os, null);
027    writer.write(datum, encoder);
028    encoder.flush();
029    return os.toByteArray();
030  }
031}
