001package org.gbif.metrics.cube.bucketer;
002
003import org.gbif.api.vocabulary.Country;
004
005import com.urbanairship.datacube.CSerializable;
006import com.urbanairship.datacube.bucketers.AbstractIdentityBucketer;
007import com.urbanairship.datacube.serializables.BytesSerializable;
008
009/**
010 * Bucketer that serializes as 2 letter ISO codes.
011 */
012public class CountryBucketer extends AbstractIdentityBucketer<Country> {
013  // The number of bytes this buckets into (2 letter ISO codes)
014  public static final int BYTES=2;
015
016  @Override
017  public CSerializable makeSerializable(Country coord) {
018    return new BytesSerializable(coord.getIso2LetterCode().getBytes());
019  }
020}
