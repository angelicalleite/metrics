001package org.gbif.metrics.cube.bucketer;
002
003import org.gbif.api.vocabulary.EndpointType;
004import org.gbif.occurrencestore.util.EndpointTypeConverter;
005
006import com.urbanairship.datacube.CSerializable;
007import com.urbanairship.datacube.bucketers.AbstractIdentityBucketer;
008import com.urbanairship.datacube.serializables.IntSerializable;
009
010/**
011 * Bucketer that serializes instances of EndpointType.
012 */
013public class EndpointTypeBucketer extends AbstractIdentityBucketer<EndpointType> {
014
015  // The number of bytes this buckets into
016  public static final int BYTES = 4;
017
018  private static final EndpointTypeConverter CONVERTER = new EndpointTypeConverter();
019
020  @Override
021  public CSerializable makeSerializable(EndpointType endpointType) {
022    return new IntSerializable(CONVERTER.fromEnum(endpointType));
023  }
024}
