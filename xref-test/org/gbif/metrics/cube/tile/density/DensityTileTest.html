63    @Test
64    public void testDownscale() {
65      Map<Layer, Map<Integer, Integer>> layers =
66        ImmutableMap.<Layer, Map<Integer, Integer>>builder()
67          .put(Layer.FOSSIL, ImmutableMap.<Integer, Integer>of(514, 1, 515, 1))
68          .build();
69      DensityTile tile1px = new DensityTile(0, 0, 0, 1, layers);
70  
71      DensityTile tile2px = tile1px.downscale(2);
72      assertEquals("Should still have only 1 layer", 1, tile2px.layers().size());
73      assertEquals("Downsizing should have merged adjacent cells", 1, tile2px.layers().get(Layer.FOSSIL).size());
74      assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
75        tile2px.layers().get(Layer.FOSSIL).get(129)); // the new key
76  
77      DensityTile tile4px = tile1px.downscale(4);
78      assertEquals("Should still have only 1 layer", 1, tile4px.layers().size());
79      assertEquals("Downsizing should have merged adjacent cells", 1, tile4px.layers().get(Layer.FOSSIL).size());
80      assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
81        tile4px.layers().get(Layer.FOSSIL).get(0)); // the new key
82  
83      // test the 2-4 downscale
84      tile4px = tile2px.downscale(4);
85      assertEquals("Should still have only 1 layer", 1, tile4px.layers().size());
86      assertEquals("Downsizing should have merged adjacent cells", 1, tile4px.layers().get(Layer.FOSSIL).size());
87      assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
88        tile4px.layers().get(Layer.FOSSIL).get(0)); // the new key
89    }

130   private final int totalCount(DensityTile tile) {
131     int total = 0;
132     for (Entry<Layer, Map<Integer, Integer>> e : tile.layers().entrySet()) {
