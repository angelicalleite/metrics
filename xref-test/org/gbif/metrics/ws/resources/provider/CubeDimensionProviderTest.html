25    // Builds an Address (package declarations necessary, as this is the INTERNAL one)
26    private Address getInternallyBuilt() {
27      return new ReadBuilder(org.gbif.metrics.cube.occurrence.OccurrenceCube.INSTANCE)
28        .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.BASIS_OF_RECORD, BasisOfRecord.OBSERVATION)
29        .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.NUB_KEY, 212)
30        .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.DATASET_KEY, DS)
31        .build();
32    }
33  
34    // An address built by the Jersey provider must be the same as a DC readbuilder version
35    @Test
36    public void testBuild() {
37      OccurrenceCubeReaderProvider b = new OccurrenceCubeReaderProvider(null);
38      MultivaluedMap<String, String> m = new MultivaluedMapImpl();
39      m.add(OccurrenceCube.BASIS_OF_RECORD.getKey(), BasisOfRecord.OBSERVATION.toString());
40      m.add(OccurrenceCube.NUB_KEY.getKey(), "212");
41      m.add(OccurrenceCube.DATASET_KEY.getKey(), DS.toString());
42      ReadBuilder rb = b.build(m);
43      assertEquals(getInternallyBuilt(), rb.build());
44    }
