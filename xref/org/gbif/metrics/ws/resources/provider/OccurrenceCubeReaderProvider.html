28  /**
29   * A provider that will convert parameters from the request into an INTERNAL cube reader.
30   */
31  @Provider
32  @Singleton
33  public class OccurrenceCubeReaderProvider implements InjectableProvider<ProvidedOccurrenceCubeReader, Parameter> {
34  
35    // Provides an index to lookup the API Dimension by the HTTP parameter key
36    private static final ImmutableMap<String, Dimension<?>> PARAM_INDEX;
37  
38    static {
39      ImmutableMap.Builder<String, Dimension<?>> b = new ImmutableMap.Builder<String, Dimension<?>>();
40      for (Dimension<?> d : org.gbif.api.model.metrics.cube.OccurrenceCube.DIMENSIONS) {
41        b.put(d.getKey(), d);
42      }
43      PARAM_INDEX = b.build();
44    }
45  
46    @Context
47    private final HttpContext hc;
48  
49    public OccurrenceCubeReaderProvider(@Context HttpContext hc) {
50      this.hc = hc;
51    }
52  
53    // The type casting is actually checked at runtime, and there _appears_ to be no way of
54    // checking the enum, hence the rawtypes and unchecked warnings suppression.
55    @SuppressWarnings({"unchecked", "rawtypes"})
56    @VisibleForTesting
57    ReadBuilder build(MultivaluedMap<String, String> params) throws ServiceUnavailableException, IllegalArgumentException {
58      ReadBuilder b = new ReadBuilder(OccurrenceCube.INSTANCE);
59      for (Entry<String, List<String>> param : params.entrySet()) {
60        // We only accept 1 value per parameter
61        String k = param.getKey();
62        String v = param.getValue().iterator().next();
63  
64        // callback is a reserved word in the API and invokes the JS callback hack
65        // JQuery also inserts _ when using the callback=?
66        if ("callback".equalsIgnoreCase(k) || "_".equalsIgnoreCase(k)) {
67          continue;
68        }
69  
70        Dimension<?> dim = PARAM_INDEX.get(k);
71        Preconditions.checkArgument(dim != null, "Unknown parameter found in the request, does not map to a dimension: "
72          + k);
73        Class<?> type = dim.getType();
74  
75        // translate the API dimension to the internal dimension, typing the value
76        // Note: The API_MAPPING might be determined at runtime, with the cube's implementing an interface allowing that
77        // to be read.
78        com.urbanairship.datacube.Dimension<?> internalDim = OccurrenceCube.API_MAPPING.get(dim);
79        if (internalDim == null) {
80          throw new ServiceUnavailableException("Unable to convert address in the cube.  No dimension exists for "
81            + dim.getKey());
82        } else if (String.class == type) {
83          b.at(internalDim, v);
84  
85        } else if (Integer.class == type) {
86          b.at(internalDim, Integer.valueOf(v));
87  
88        } else if (Float.class == type) {
89          b.at(internalDim, Float.valueOf(v));
90  
91        } else if (Double.class == type) {
92          b.at(internalDim, Double.valueOf(v));
93  
94        } else if (Boolean.class == type) {
95          b.at(internalDim, Boolean.valueOf(v));
96  
97        } else if (UUID.class == type) {
98          b.at(internalDim, UUID.fromString(v));
99  
100       } else if (Enum.class.isAssignableFrom(type)) {
101         b.at(internalDim, Enum.valueOf((Class<Enum>) dim.getType(), v));
102 
103       } else {
104         throw new ServiceUnavailableException("Dimension is of unknown type: " + type);
105       }
106     }
107     return b;
108   }
