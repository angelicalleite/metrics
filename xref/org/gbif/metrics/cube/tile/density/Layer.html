46    /**
47     * Utility to determine to which layer an occurrence record belongs.
48     * This assumes that the record is plottable, so no verification on the coordinates is performed.
49     * 
50     * @param bor The basis of record for the Occurrence
51     * @param year The year for the Occurrence
52     * @return The layer that a record should belong to.
53     */
54    public static Layer inferFrom(BasisOfRecord bor, Integer year) {
55      switch (bor) {
56        case FOSSIL_SPECIMEN:
57          return FOSSIL;
58        case LIVING_SPECIMEN:
59          return LIVING;
60  
61        case PRESERVED_SPECIMEN:
62          return inferYear(year, SPECIMENS);
63  
64        case OBSERVATION:
65        case HUMAN_OBSERVATION:
66        case MACHINE_OBSERVATION:
67          return inferYear(year, OBSERVATIONS);
68  
69        default:
70          return inferYear(year, OTHERS);
71      }
72    }

74    /**
75     * Utility to determine to which layer an occurrence record belongs.
76     * This assumes that the record is plottable, so no verification on the coordinates is performed.

78     * @param o The Occurrence record which needs to be placed.
79     * @return The layer that a record should belong to.

81    public static Layer inferFrom(OccurrenceWritable o) {

86    /**
87     * Given the choices which are of a careful ordering (see definitions above) returns the
88     * correct year.
89     */
90    private static Layer inferYear(Integer year, Layer[] choices) {
91      if (year == null) {
92        return choices[0];
93      }
94  
95      if (year < 1900) {
96        return choices[1];
97      }
98  
99      int index = 2;
100     for (int decade = 1900; decade <= 2010; decade += 10, index += 1) {
101       if (year >= decade && year < decade + 10) {
102         return choices[index];
103       }
104     }
105 
106     // could be post 2020 for example
107     return choices[0];
108   }
