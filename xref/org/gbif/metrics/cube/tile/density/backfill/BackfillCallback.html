27  /**
28   * The callback used from the backfill process to spawn the job to write the new data in the cube.
29   * This runs 2 MR jobs:
30   * i) Group the occurrence records at the same location with the same identification in the same dataset etc.
31   * ii) Build the cube for all the dimensions.
32   * MR job i) is purely to reduce load, as MR job ii) emits several tiles at each zoom level (= a lot to shuffle and
33   * sort)
34   */
35  class BackfillCallback implements HBaseBackfillCallback {

37    private void addFieldToScan(Scan scan, FieldName fn) {
38      scan.addColumn(Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnFamilyName()),
39        Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnName()));
40    }

42    @Override
43    public void backfillInto(Configuration conf, byte[] table, byte[] cf, long snapshotFinishMs) throws IOException {
44      conf = HBaseConfiguration.create(conf);
45      conf.set("mapred.reduce.slowstart.completed.maps", "1.0"); // delay reducers to spare the machine load
46  
47      // a temporary directory for using to store intermediate data between the 2 jobs
48      String dir = UUID.randomUUID().toString();
49      Path tmpDir = new Path("/tmp/" + dir);
50  
51      try {
52        // JOB #1 grouping the occurrences
53        Job job = new Job(conf, "density-cube backfill prepare");
54        job.setJarByClass(TileCollectorMapper.class); // required to set up MR classpath
55        job.getConfiguration().set("mapred.task.timeout", "600000"); // 10 mins
56        job.getConfiguration().set("hbase.regionserver.lease.period", "600000");
57        job.getConfiguration().set("mapred.compress.map.output", "true");
58        job.getConfiguration().set("mapred.output.compress", "true");
59        TableMapReduceUtil.initTableMapperJob(conf.get(Backfill.KEY_SOURCE_TABLE), getScanner(conf),
60          LocationCollectorMapper.class, OccurrenceWritable.class, IntWritable.class, job);
61        job.setCombinerClass(GroupByOccurrenceReducer.class); // Optimize network bandwidth
62        job.setReducerClass(GroupByOccurrenceReducer.class);
63        job.setNumReduceTasks(conf.getInt(Backfill.KEY_NUM_REDUCERS, Backfill.DEFAULT_NUM_REDUCERS));
64        job.setOutputKeyClass(OccurrenceWritable.class);
65        job.setOutputValueClass(IntWritable.class);
66        job.setOutputFormatClass(SequenceFileOutputFormat.class);
67        SequenceFileOutputFormat.setOutputPath(job, tmpDir);
68        if (!job.waitForCompletion(true)) {
69          throw new IOException("Unknown error with job.  Check the logs.");
70        }
71  
72        // Job #2 to write the cube
73        conf.set("mapred.child.java.opts", "-Xmx3G"); // for safety, do not assume the default works
74        job = new Job(conf, "density-cube backfill write");
75        job.setJarByClass(TileCollectorMapper.class); // required to set up MR classpath
76        job.setNumReduceTasks(conf.getInt(Backfill.KEY_NUM_REDUCERS, Backfill.DEFAULT_NUM_REDUCERS));
77        SequenceFileInputFormat.setInputPaths(job, tmpDir);
78        // This appears to be the only way to get a SequenceFileInputFormat to understand types
79        SequenceFileInputFormat<OccurrenceWritable, IntWritable> sequenceInputFormat =
80          new SequenceFileInputFormat<OccurrenceWritable, IntWritable>();
81        job.setInputFormatClass(sequenceInputFormat.getClass());
82        job.setMapperClass(TileCollectorMapper.class);
83        job.setMapOutputKeyClass(TileKeyWritable.class);
84        job.setMapOutputValueClass(OccurrenceWritable.class);
85        // Ensure no bad counts in the cube
86        job.getConfiguration().set("mapred.map.tasks.speculative.execution", "false");
87        job.getConfiguration().set("mapred.reduce.tasks.speculative.execution", "false");
88        job.getConfiguration().set("mapred.compress.map.output", "true");
89        job.setReducerClass(CubeWriterReducer.class);
90        job.setOutputFormatClass(NullOutputFormat.class);
91        if (!job.waitForCompletion(true)) {
92          throw new IOException("Unknown error with job.  Check the logs.");
93        }
94      } catch (Exception e) {
95        throw new IOException(e);
96      } finally {
97        // Cleanup by deleting the temporary directory
98        FileSystem fs = null;
99        try {
100         fs = FileSystem.get(tmpDir.toUri(), conf);
101         fs.delete(tmpDir, true);
102       } finally {
103         fs.close();
104       }
105     }
106   }

108   private Scan getScanner(Configuration conf) {

111     scan.setCacheBlocks(false); // not needed for efficient scanning
