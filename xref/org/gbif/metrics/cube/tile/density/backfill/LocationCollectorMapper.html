18  /**
19   * Reads the HBase table, collecting the terms we are interested in as we know there are duplicates
20   * at the same location. This is only to reduce the amount of grouping needed in later MR jobs.
21   */
22  public class LocationCollectorMapper extends TableMapper<OccurrenceWritable, IntWritable> {
23  
24    private final static IntWritable ONE = new IntWritable(1);
25    private final BasisOfRecordConverter borc = new BasisOfRecordConverter();
26  
27    /**
28     * Reads the table, emits the Result keyed on the Lat Lng if it is a plottable record
29     */
30    @Override
31    protected void map(ImmutableBytesWritable key, Result row, Context context) throws IOException, InterruptedException {
32  
33      Double latitude = OccurrenceResultReader.getDouble(row, FieldName.I_LATITUDE);
34      Double longitude = OccurrenceResultReader.getDouble(row, FieldName.I_LONGITUDE);
35  
36      // Google only goes +/- 85 degrees and we only want maps with no known issues
37      if (!OccurrenceWritable.hasSpatialIssue(row) && MercatorProjectionUtil.isPlottable(latitude, longitude)) {
38  
39        // Note: Make sure everything read here is in the getScanner() in BackFillCallback!
40        Integer kingdomID = OccurrenceResultReader.getInteger(row, FieldName.I_KINGDOM_ID);
41        Integer phylumID = OccurrenceResultReader.getInteger(row, FieldName.I_PHYLUM_ID);
42        Integer classID = OccurrenceResultReader.getInteger(row, FieldName.I_CLASS_ID);
43        Integer orderID = OccurrenceResultReader.getInteger(row, FieldName.I_ORDER_ID);
44        Integer familyID = OccurrenceResultReader.getInteger(row, FieldName.I_FAMILY_ID);
45        Integer genusID = OccurrenceResultReader.getInteger(row, FieldName.I_GENUS_ID);
46        Integer speciesID = OccurrenceResultReader.getInteger(row, FieldName.I_SPECIES_ID);
47        // taxon != species (it may be a higher taxon, or might be a subspecies)
48        Integer taxonID = OccurrenceResultReader.getInteger(row, FieldName.I_NUB_ID);
49        String publishingOrganisationKey = OccurrenceResultReader.getString(row, FieldName.OWNING_ORG_KEY);
50        String datasetKey = OccurrenceResultReader.getString(row, FieldName.DATASET_KEY);
51        String countryIsoCode = OccurrenceResultReader.getString(row, FieldName.I_ISO_COUNTRY_CODE);
52        String hostCountryIsoCode = OccurrenceResultReader.getString(row, FieldName.HOST_COUNTRY);
53        Integer year = OccurrenceResultReader.getInteger(row, FieldName.I_YEAR);
54  
55        Integer borAsInt = OccurrenceResultReader.getInteger(row, FieldName.I_BASIS_OF_RECORD);
56        BasisOfRecord bor = borc.toEnum(borAsInt);
57        EndpointType protocol = EndpointType.valueOf(OccurrenceResultReader.getString(row, FieldName.PROTOCOL));
58  
59        context.getCounter(bor).increment(1);
60  
61        context.write(new OccurrenceWritable(kingdomID, phylumID, classID, orderID, familyID, genusID, speciesID,
62          taxonID, false, publishingOrganisationKey, datasetKey, countryIsoCode, hostCountryIsoCode, latitude,
63          longitude, year, null, // month
64          bor, protocol, 1), ONE);
65      }
66    }
