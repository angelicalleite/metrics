15     * @return the X pixel offset within the tile
16     */
17    public static int getOffsetX(double lat, double lng, int zoom) {
18      double x = toNormalisedPixelCoords(lat, lng).getX();
19      int scale = 1 << zoom;
20      x *= scale * TILE_SIZE;
21      return (int) (x % TILE_SIZE);

26     * @return the Y pixel offset within the tile.
27     */
28    public static int getOffsetY(double lat, double lng, int zoom) {
29      double y = toNormalisedPixelCoords(lat, lng).getY();
30      int scale = 1 << zoom;
31      y *= scale * TILE_SIZE;
32      int tileY = toTileY(lat, zoom);
33      return (int) (y - (TILE_SIZE * tileY));
34    }

36    /**
37     * @return a Rectangle2D with x = lon, y = lat, width=lonSpan, height=latSpan
38     *         for an x,y,zoom as used by google.
39     */
40    public static Rectangle2D.Double getTileRect(int x, int y, int zoom) {
41      int tilesAtThisZoom = 1 << (17 - zoom);
42      double lngWidth = 360.0 / tilesAtThisZoom; // width in degrees longitude
43      double lng = -180 + (x * lngWidth); // left edge in degrees longitude
44  
45      double latHeightMerc = 1.0 / tilesAtThisZoom; // height in "normalized" mercator 0,0 top left
46      double topLatMerc = y * latHeightMerc; // top edge in "normalized" mercator 0,0 top left
47      double bottomLatMerc = topLatMerc + latHeightMerc;
48  
49      // convert top and bottom lat in mercator to degrees
50      // note that in fact the coordinates go from about -85 to +85 not -90 to 90!
51      double bottomLat = Math.toDegrees((2 * Math.atan(Math.exp(Math.PI * (1 - (2 * bottomLatMerc))))) - (Math.PI / 2));
52  
53      double topLat = Math.toDegrees((2 * Math.atan(Math.exp(Math.PI * (1 - (2 * topLatMerc))))) - (Math.PI / 2));
54  
55      double latHeight = topLat - bottomLat;
56  
57      return new Rectangle2D.Double(lng, bottomLat, lngWidth, latHeight);
58    }

65    public static boolean isPlottable(Double lat, Double lng) {

69    /**
70     * Returns the lat/lng as an "Offset Normalized Mercator" pixel coordinate,
71     * this is a coordinate that runs from 0..1 in latitude and longitude with 0,0 being
72     * top left. Normalizing means that this routine can be used at any zoom level and
73     * then multiplied by a power of two to get actual pixel coordinates.
74     */
75    public static Point2D toNormalisedPixelCoords(double lat, double lng) {
76      // first convert to Mercator projection
77      // first convert the lat lon to mercator coordintes.
78      if (lng > 180) {
79        lng -= 360;
80      }
81  
82      lng /= 360;
83      lng += 0.5;
84  
85      lat = 0.5 - ((Math.log(Math.tan((Math.PI / 4) + ((0.5 * Math.PI * lat) / 180))) / Math.PI) / 2.0);
86  
87      return new Point2D.Double(lng, lat);
88    }

91     * @return The tile index for the longitude at the given zoom
92     */
93    public static int toTileX(double lng, int zoom) {
94      if (lng > 180) {
95        lng -= 360;
96      }
97  
98      lng /= 360;
99      lng += 0.5;
100     int scale = 1 << zoom;
101     return (int) (lng * scale);
102   }

105   /**
106    * Returns a point that is a google tile reference for the tile containing
107    * the lat/lng and at the zoom level.
108    */
109   public static Point toTileXY(double lat, double lng, int zoom) {
110     Point2D normalised = toNormalisedPixelCoords(lat, lng);
111     int scale = 1 << zoom;
112 
113     // can just truncate to integer, this looses the fractional "pixel offset"
114     return new Point((int) (normalised.getX() * scale), (int) (normalised.getY() * scale));
115   }
