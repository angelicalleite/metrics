8   /**
9    * Accumulates the counts of occurrences, but DOES NOT set the count on the occurrence itself.
10   * This is required so that it can have several passes, and the keys remain the same.
11   */
12  public class OccurrenceWritableCombiner extends Reducer<OccurrenceWritable, IntWritable, OccurrenceWritable, IntWritable> {
13  
14    private final IntWritable i = new IntWritable();
15  
16    @Override
17    protected void reduce(OccurrenceWritable key, Iterable<IntWritable> value, Context context) throws IOException,
18      InterruptedException {
19      int total = 0;
20      for (IntWritable i : value) {
21        total += i.get();
22      }
23      context.setStatus("Occurrence from dataset[" + key.getDatasetKey() + "],  Taxon[" + key.getTaxonID() + "], Count["
24        + total + "]");
25      i.set(total);
26      context.write(key, i);
27    }
