27  /**
28   * A utility class to encapsulate the logic needed to convert between the denormalized
29   * format of occurrences (which have kingdomKey, phylumKey etc) and the normalized format
30   * used by the occurrence cube (which uses only a nubKey).
31   * This class uses the occurrence cube rollup definitions to determine which cube addresses
32   * are to be updated for a given record. This is only necessary due to the normalization
33   * that is required to for the taxonomic ranks that are all merged into a single nub key.
34   * Brief explanation:
35   * - Imagine a record from dataset:1 with kingdom:1 and phylum:2
36   * - Imagine the cube counts "by dataset", "records by nubKey" and "records by dataset by nubKey"
37   * We need to update the following cube addresses:
38   * - "dataset:1"
39   * - "nubKey:1"
40   * - "nubKey:2"
41   * - "dataset:1,nubKey:1"
42   * - "dataset:1,nubKey:2"
43   * You cannot update the cube 2 times for each taxon, or else the dataset count will be 2 in this
44   * example.
45   * This is the problem this class addresses, by providing the addresses for a single occurrence record.
46   */
47  public class OccurrenceAddressUtil {

49    /**
50     * For the given occurrence, determines the mutations (addresses and operations) that need
51     * to be applied.
52     *
53     * @param occurrence The denormalized representation
54     * @param op That is going to be applied to the cube
55     * @return The batch of updates to apply
56     */
57    public static Batch<LongOp> cubeMutation(Occurrence occurrence, LongOp op) {
58      // Needs mutability, since we rely on overwriting as a lazy distinction mechanism
59      // such as for when speciesKey = nubKey
60      Map<Address, LongOp> m = Maps.newHashMap();
61      m.putAll(mutationsForTaxon(occurrence, occurrence.getKingdomKey(), op));
62      m.putAll(mutationsForTaxon(occurrence, occurrence.getPhylumKey(), op));
63      m.putAll(mutationsForTaxon(occurrence, occurrence.getClassKey(), op));
64      m.putAll(mutationsForTaxon(occurrence, occurrence.getOrderKey(), op));
65      m.putAll(mutationsForTaxon(occurrence, occurrence.getFamilyKey(), op));
66      m.putAll(mutationsForTaxon(occurrence, occurrence.getGenusKey(), op));
67      m.putAll(mutationsForTaxon(occurrence, occurrence.getSpeciesKey(), op));
68      m.putAll(mutationsForTaxon(occurrence, occurrence.getTaxonKey(), op));
69      return new Batch<LongOp>(m);
70    }

95    /**
96     * For the given taxon, returns all the updates to be applied to the cube
97     */
98    private static Map<Address, LongOp> mutationsForTaxon(Occurrence occurrence, Integer nubKey, LongOp op) {
99      WriteBuilder wb = new WriteBuilder(OccurrenceCube.INSTANCE);
100     addEnumDimension(wb, occurrence.getCountry(), COUNTRY);
101     addEnumDimension(wb, occurrence.getProtocol(), PROTOCOL);
102     addGeoreferencingDimension(wb, occurrence, IS_GEOREFERENCED);
103     addEnumDimension(wb, occurrence.getBasisOfRecord(), BASIS_OF_RECORD);
104     addUUIDDimension(wb, occurrence.getDatasetKey(), DATASET_KEY);
105     addEnumDimension(wb, occurrence.getPublishingCountry(), HOST_COUNTRY);
106     addIntDimension(wb, nubKey, NUB_KEY);
107     addIntDimension(wb, occurrence.getYear(), YEAR);
108     return OccurrenceCube.INSTANCE.getWrites(wb, op).getMap();
109   }

114   private static Map<Address, LongOp> mutationsForTaxon(OccurrenceWritable occurrence, Integer nubKey, LongOp op) {
115     WriteBuilder wb = new WriteBuilder(OccurrenceCube.INSTANCE);
116     addCountryEnumDimension(wb, occurrence.getCountryIsoCode(), COUNTRY);
117     addEnumDimension(wb, occurrence.getProtocol(), PROTOCOL);
118     addGeoreferencingDimension(wb, occurrence, IS_GEOREFERENCED);
119     addEnumDimension(wb, occurrence.getBasisOfRecord(), BASIS_OF_RECORD);
120     addUUIDDimension(wb, occurrence.getDatasetKey(), DATASET_KEY);
121     addCountryEnumDimension(wb, occurrence.getHostCountryIsoCode(), HOST_COUNTRY);
122     addIntDimension(wb, nubKey, NUB_KEY);
123     addIntDimension(wb, occurrence.getYear(), YEAR);
124     return OccurrenceCube.INSTANCE.getWrites(wb, op).getMap();
125   }
