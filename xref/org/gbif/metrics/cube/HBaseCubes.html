34  /**
35   * This is a utility class to simplify dealing with the HBase CubeIo IO.
36   * Depending on usage, sometimes you need access to the underlying Pool of HTables, to control for example auto
37   * flushing. This class is intended to simplify those operations by providing a simpler API than those provided
38   * by the underlying HBase and DataCube libraries.
39   * There are 2 key scenarios typically you want to run DataCube in:
40   * <ol>
41   * <li>In a backfill to populate a new table, where certain optimizations can be made</li>
42   * <li>In live read write mode</li>
43   * </ol>
44   * Note: Decision was to use separate table per cube and per lookup (to enable easy truncating by cube), hence
45   * uniqueCubeName is empty in the idService and the hbaseHarness. This is a decision that might be revisited in the
46   * future.
47   */
48  public class HBaseCubes<O extends Op> {

68    private HBaseCubes(DataCube<O> cube, Deserializer<O> deserializer, byte[] cubeTable, byte[] lookupTable,
69      byte[] counterTable, byte[] cf, Configuration conf, int writeBatchSize, CommitType commitType, SyncLevel syncLevel,
70      HTableFactory factory, int numFlushThreads, int numIoeRetries, int numCasRetries) throws IOException {
71      this.cubeTable = cubeTable;
72      this.syncLevel = syncLevel;
73      pool = new HTablePool(conf, 10, factory);
74      IdService idService = new HBaseIdService(conf, lookupTable, counterTable, cf, EMPTY_BYTE_ARRAY);
75      DbHarness<O> hbaseDbHarness =
76        new HBaseDbHarness<O>(pool, EMPTY_BYTE_ARRAY, cubeTable, cf, deserializer, idService, commitType,
77          numFlushThreads, numIoeRetries, numCasRetries, null); // these are the defaults
78      // Note: batches are flushed after 1 minute
79      dataCubeIo = new DataCubeIo<O>(cube, hbaseDbHarness, writeBatchSize, TimeUnit.MINUTES.toMillis(1), syncLevel);
80    }

189   // Utility to create an HTableFactory that disables auto flushing
190   private static HTableFactory newNonFlushingHTableFactory() {
191     return new HTableFactory() {
192 
193       @Override
194       public HTableInterface createHTableInterface(Configuration config, byte[] tableName) {
195         try {
196           HTable table = new HTable(config, tableName);
197           table.setAutoFlush(false);
198           return table;
199         } catch (IOException ioe) {
200           throw new RuntimeException(ioe);
201         }
202       }
203     };
204   }

231   /**
232    * Closes the underlying connections, flushing etc. Following this no more operations will function.
233    */
234   public void close() throws IOException, InterruptedException {
235     try {
236       if (pool != null) {
237         flush();
238         pool.closeTablePool(cubeTable);
239       }
240     } catch (NullPointerException e) {
241       // thrown when the pool has never issued this table, indicating an error
242       throw new IOException("Attempt to close HBase pool, when table has not been written to.");
243     }
244   }
245 
246   /**
247    * Closes the underlying connections, flushing etc. Following this no more operations will function.
248    */
249   public void closeQuietly() {
250     try {
251       close();
252     } catch (Exception e) {
253       LOG.warn("Attempt to close HBase pool, when table has not been written to.  Ignoring");
254     }
255   }

271   /**
272    * Utility to write a batch to the cube.
273    * NOTE: This is for advanced users only, and requires an understanding of DataCube internals.
274    * For example, one must know that this ONLY works for the Batch_Async mode.
275    * 
276    * @param batch To perform
277    * @throws IllegalStateException Should this be called with the wrong sync level
278    */
279   public void write(Batch<O> batch) throws IllegalStateException, AsyncException, InterruptedException {
280     Preconditions.checkState(SyncLevel.BATCH_ASYNC == syncLevel);
281     dataCubeIo.writeAsync(batch);
282   }

284   /**
285    * Utility to simplify writes, by calling the appropriate write or writeAsync
286    * 
287    * @throws AsyncException only if using the ASync mode
288    */
289   public void write(O op, WriteBuilder wb) throws AsyncException, InterruptedException, IOException {
290     if (SyncLevel.BATCH_ASYNC == syncLevel) {
291       dataCubeIo.writeAsync(op, wb);
292     } else {
293       dataCubeIo.writeSync(op, wb);
294     }
