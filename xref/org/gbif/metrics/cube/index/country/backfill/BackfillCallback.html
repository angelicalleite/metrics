21  /**
22   * The callback used from the backfill process to spawn the job to write the new
23   * data in the cube. This is a simple 1 pass scan of the occurrence data which
24   * writes the cube.
25   */
26  class BackfillCallback implements HBaseBackfillCallback {
27  
28    @Override
29    public void backfillInto(Configuration conf, byte[] table, byte[] cf, long snapshotFinishMs) throws IOException {
30      conf = HBaseConfiguration.create(conf);
31  
32      try {
33        Job job = new Job(conf, "Dataset by country cube writer");
34        job.setJarByClass(TableReaderMapper.class); // required to set up MR classpath
35        // These are set high, as any failure means inaccurate cube counts
36        job.getConfiguration().set("mapred.task.timeout", "1800000"); // 30 mins
37        job.getConfiguration().set("hbase.regionserver.lease.period", "1800000");
38        // We are doing a huge amount of very tiny PUTs so reduce the buffer, as
39        // the default of 2MB means HBase
40        // is being asked to process huge numbers of rows, and tasks start to
41        // timeout. These will likely hit
42        // 1 region if the cube is initially empty.
43        job.getConfiguration().set("hbase.client.write.buffer", "262144"); // 256k
44  
45        TableMapReduceUtil.initTableMapperJob(conf.get(HBaseSourcedBackfill.KEY_SOURCE_TABLE), getScanner(conf),
46          TableReaderMapper.class, Key.class, IntWritable.class, job);
47        job.setCombinerClass(Combiner.class); // to reduce transfer, combine on
48                                              // the map emission
49        job.setReducerClass(CubeWriterReducer.class);
50        job.setNumReduceTasks(conf.getInt(HBaseSourcedBackfill.KEY_NUM_REDUCERS,
51          HBaseSourcedBackfill.DEFAULT_NUM_REDUCERS));
52        job.setOutputKeyClass(NullWritable.class);
53        job.setOutputValueClass(NullWritable.class);
54        job.setOutputFormatClass(NullOutputFormat.class);
55        if (!job.waitForCompletion(true)) {
56          throw new IOException("Unknown error with job.  Check the logs.");
57        }
58  
59      } catch (Exception e) {
60        throw new IOException(e);
61      }
62    }
63  
64    private void addFieldToScan(Scan scan, FieldName fn) {
65      scan.addColumn(Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnFamilyName()),
66        Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fn).getColumnName()));
67    }
68  
69    private Scan getScanner(Configuration conf) {
70      Scan scan = new Scan();
71      scan.setCaching(conf.getInt(HBaseSourcedBackfill.KEY_SCANNER_CACHE, HBaseSourcedBackfill.DEFAULT_SCANNER_CACHE));
72      scan.setCacheBlocks(false); // not needed for efficient scanning
73      // Optimize the scan by bringing back only what the TableReaderMapper wants
74      addFieldToScan(scan, FieldName.I_ISO_COUNTRY_CODE);
75      addFieldToScan(scan, FieldName.DATASET_KEY);
76      return scan;
77    }
