23    private void insertIfPresent(Set<Integer> t, Integer v) {
24      if (v != null) {
25        t.add(v);
26      }
27    }

29    @Override
30    protected void map(ImmutableBytesWritable key, Result row, Context context) throws IOException, InterruptedException {
31      try {
32        Occurrence o = OccurrenceBuilder.buildOccurrence(row);
33        Preconditions.checkNotNull(o, "Unable to generate an Occurrence from the HBase using the OccurrenceBuilder");
34  
35        if (o.getDatasetKey() != null) {
36          Set<Integer> nubs = Sets.newHashSet();
37          insertIfPresent(nubs, o.getKingdomKey());
38          insertIfPresent(nubs, o.getPhylumKey());
39          insertIfPresent(nubs, o.getClassKey());
40          insertIfPresent(nubs, o.getOrderKey());
41          insertIfPresent(nubs, o.getFamilyKey());
42          insertIfPresent(nubs, o.getGenusKey());
43          insertIfPresent(nubs, o.getSubgenusKey());
44          insertIfPresent(nubs, o.getSpeciesKey());
45          insertIfPresent(nubs, o.getTaxonKey());
46          for (Integer i : nubs) {
47            Key k = new Key(o.getDatasetKey(), i);
48            context.setStatus(k.toString());
49            context.write(k, ONE);
50          }
51        } else {
52          context.getCounter("GBIF", "Missing dataset key (skipped record)").increment(1);
53        }
54      } catch (Exception e) {
55        throw new IOException(e);
56      }
57    }
