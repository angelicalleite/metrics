19  /**
20   * A reducer that sums the counts and then writes to the cube.
21   */
22  public class CubeWriterReducer extends Reducer<Key, IntWritable, NullWritable, NullWritable> {
23  
24    private HBaseCubes<UuidIntMap> hbaseCube;
25  
26    @Override
27    protected void cleanup(Context context) throws IOException, InterruptedException {
28      super.cleanup(context);
29      hbaseCube.close();
30    }
31  
32    /**
33     * Writes to the cube.
34     */
35    @Override
36    protected void reduce(Key key, Iterable<IntWritable> values, Context context) throws IOException {
37      int total = 0;
38      for (IntWritable i : values) {
39        total += i.get();
40      }
41      context.setStatus("Dataset[" + key.getDatasetKey() + "],  NubKey[" + key.getNubKey() + "], Count[" + total + "]");
42      try {
43        UuidIntMap c = UuidIntMap.newInstance(ImmutableMap.<UUID, Integer>of(key.getDatasetKey(), total));
44        WriteBuilder wb =
45          new WriteBuilder(OccurrenceDatasetCube.INSTANCE).at(OccurrenceDatasetCube.NUB_KEY, key.getNubKey());
46        hbaseCube.write(c, wb);
47  
48      } catch (Exception e) {
49        throw new IOException(e);
50      }
51    }
52  
53    @Override
54    protected void setup(Context context) throws IOException, InterruptedException {
55      super.setup(context);
56      Configuration conf = context.getConfiguration();
57      int writeBatchSize =
58        conf.getInt(HBaseSourcedBackfill.KEY_WRITE_BATCH_SIZE, HBaseSourcedBackfill.DEFAULT_WRITE_BATCH_SIZE);
59      hbaseCube =
60        HBaseCubes.newCombiningBatchAsync(OccurrenceDatasetCube.INSTANCE, UuidIntMap.DESERIALIZER,
61          Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_BACKFILL_TABLE)),
62          Bytes.toBytes(conf.get(HBaseSourcedBackfill.KEY_CF)),
63          conf, writeBatchSize);
64    }
