22  /**
23   * Base class to simplify writing message callbacks that mutate cubes by providing the message handling and the timing
24   * services.
25   *
26   * @param <T> The cube type
27   */
28  abstract class CubeUpdaterCallback<T extends Op> extends AbstractMessageCallback<OccurrenceMutatedMessage> {
29  
30    private static final Logger LOG = LoggerFactory.getLogger(CubeUpdaterCallback.class);
31    private final Counter messageCount = Metrics.newCounter(getClass(), "messageCount");
32    private final Counter newOccurrencesCount = Metrics.newCounter(getClass(), "newOccurrencesCount");
33    private final Counter updatedOccurrencesCount = Metrics.newCounter(getClass(), "updatedOccurrencesCount");
34    private final Counter deletedOccurrencesCount = Metrics.newCounter(getClass(), "deletedOccurrencesCount");
35    private final Timer writeTimer = Metrics.newTimer(getClass(), "cubeWrites", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
36    private InactivePeriodFlusher inactivePeriodFlusher;
37  
38    abstract HBaseCubes<T> getCube();
39  
40    protected abstract Batch<T> getNewMutations(Occurrence occurrence);
41  
42    protected abstract Batch<T> getUpdateMutations(Occurrence oldOccurrence, Occurrence newOccurrence);
43  
44    @Override
45    public void handleMessage(OccurrenceMutatedMessage message) {
46      LOG.debug("Handling {} occurrence", message.getStatus());
47      messageCount.inc();
48      resetInactiveWatcher();
49  
50      Batch<T> update = new Batch<T>();
51      switch (message.getStatus()) {
52        case NEW:
53          newOccurrencesCount.inc();
54          update = getNewMutations(message.getNewOccurrence());
55          break;
56        case UPDATED:
57          updatedOccurrencesCount.inc();
58          // if there's no interesting difference btw occurrences, ignore this msg
59          if (OccurrenceComparisonUtil.equivalent(message.getOldOccurrence(), message.getNewOccurrence())) {
60            LOG.debug("Ignoring update on occurrence [{}] because cube related fields haven't changed.",
61              message.getOldOccurrence().getKey());
62            return;
63          }
64          update = getUpdateMutations(message.getOldOccurrence(), message.getNewOccurrence());
65          break;
66        case DELETED:
67          deletedOccurrencesCount.inc();
68          update = getUpdateMutations(message.getOldOccurrence(), null);
69          break;
70        case UNCHANGED:
71          return;
72      }
73  
74      TimerContext context = writeTimer.time();
75      try {
76        if (update.getMap() != null && !update.getMap().isEmpty()) {
77          LOG
78            .debug("Mutations for {} occurrence produced {} cube mutations", message.getStatus(), update.getMap().size());
79          getCube().write(update);
80        } else {
81          LOG.debug("No mutations for {} occurrence produced. Old[{}] New[{}], ", message.getStatus(),
82            message.getOldOccurrence(), message.getNewOccurrence());
83        }
84      } catch (AsyncException e) {
85        LOG.error("Exception while updating cube for [{}] Occurrence [{}]", message.getStatus(), e);
86      } catch (InterruptedException e) {
87        LOG.error("Interrupted while updating cube for [{}] Occurrence [{}]", message.getStatus(), e);
88      } finally {
89        context.stop();
90      }
91      LOG.debug("Finished handling {} occurrence", message.getStatus());
92    }
93  
94    protected void resetInactiveWatcher() {
95      if (inactivePeriodFlusher != null) {
96        inactivePeriodFlusher.reset();
97      }
98    }
99  
100   /**
101    * Starts an inactive watcher if there is not one already running.
102    */
103   public synchronized void startInactiveWatcher() {
104     if (inactivePeriodFlusher == null) {
105       LOG.info("Starting watcher thread to flush cube on periods of inactivity");
106       inactivePeriodFlusher = new InactivePeriodFlusher(getCube());
107       Thread t = new Thread(inactivePeriodFlusher);
108       // TODO: if we were to do the following, a flush might be terminated on shutdown
109       // however, will it block shutdown otherwise?
110       // t.setDaemon(true); // we're just a background thread
111       t.start();
112     }
113   }
