19  /**
20   * The service to update the occurrence cube.
21   */
22  class OccurrenceCubeUpdaterService extends CubeUpdaterService<CubeConfiguration> {
23    // reduce reuse recycle HBase connections
24    private volatile HBaseCubes<LongOp> singletonCube;
25  
26    public OccurrenceCubeUpdaterService(CubeConfiguration configuration) {
27      super(configuration);
28    }
29  
30    /**
31     * @return The same batch with zero values omitted
32     */
33    @VisibleForTesting
34    static Batch<LongOp> trimZeros(Batch<LongOp> batch) {
35      Iterator<Map.Entry<Address, LongOp>> iter = batch.getMap().entrySet().iterator();
36      while (iter.hasNext()) {
37        Map.Entry<Address, LongOp> entry = iter.next();
38        if (0 == entry.getValue().getLong()) {
39          iter.remove();
40        }
41      }
42      return batch;
43    }
44  
45    @Override
46    CubeUpdaterCallback<LongOp>
47      getCallback(final CubeConfiguration configuration,
48      final Configuration hadoopConfiguration) {
49  
50      return new CubeUpdaterCallback<LongOp>() {
51  
52        @Override
53        synchronized HBaseCubes<LongOp> getCube() {
54          if (singletonCube == null) {
55            try {
56              singletonCube =
57                HBaseCubes.newIncrementingBatchAsync(OccurrenceCube.INSTANCE, LongOp.DESERIALIZER,
58                  configuration.cubeTable.getBytes(), configuration.columnFamily.getBytes(), hadoopConfiguration,
59                  configuration.batchSize, configuration.batchFlushThreads);
60            } catch (IOException e) {
61              throw Throwables.propagate(e);
62            }
63          }
64          return singletonCube;
65        }
66  
67        @Override
68        protected Batch<LongOp> getNewMutations(Occurrence occurrence) {
69          return OccurrenceAddressUtil.cubeMutation(occurrence, new LongOp(1));
70        }
71  
72        @Override
73        protected Batch<LongOp> getUpdateMutations(Occurrence oldOccurrence, Occurrence newOccurrence) {
74          Batch<LongOp> update = OccurrenceAddressUtil.cubeMutation(oldOccurrence, new LongOp(-1));
75          // newOccurrence will be null for deletes
76          if (newOccurrence != null) {
77            update.putAll(OccurrenceAddressUtil.cubeMutation(newOccurrence, new LongOp(1)));
78          }
79          // adding a 1 to a -1 is common, and we know that is 0, so let's flush that out now
80          return trimZeros(update);
81        }
82      };
83    }
84  
85    @Override
86    protected void shutDown() throws Exception {
87      if (singletonCube != null) {
88        singletonCube.close();
89      }
90    }
