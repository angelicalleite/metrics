001package org.gbif.metrics.ws.client;
002
003import org.gbif.api.model.metrics.cube.OccurrenceCube;
004import org.gbif.api.model.metrics.cube.ReadBuilder;
005import org.gbif.api.service.metrics.CubeService;
006import org.gbif.metrics.ws.client.guice.MetricsWsClientModule;
007import org.gbif.metrics.ws.guice.MetricsWsServletListener;
008import org.gbif.utils.file.properties.PropertiesUtil;
009import org.gbif.ws.client.BaseResourceTest;
010import org.gbif.ws.client.guice.UrlBindingModule;
011
012import java.io.IOException;
013import java.util.UUID;
014
015import com.google.inject.Guice;
016import com.google.inject.Injector;
017import org.junit.Before;
018import org.junit.Ignore;
019import org.junit.Test;
020
021import static org.junit.Assert.assertTrue;
022
023/**
024 * A simple Grizzly based WS IT test 
025 */
026public class CubeWsClientIT extends BaseResourceTest {
027  private static final String PROPERTIES_FILE = "metrics.properties";
028  private static final String CONTEXT = "metrics";
029  // should this change then we are really in trouble
030  private static final int ANIMALIA_KEY = 1;
031  private CubeService wsClient; // under test
032  
033  public CubeWsClientIT() {
034    super("org.gbif.metrics.ws.resources", CONTEXT, MetricsWsServletListener.class);
035  }
036  
037  @Before
038  public void init() throws IOException {
039    // This uses UrlBindingModule to dynamically set the named property (metrics.ws.url) taking into account that the 
040    // port might have been set at runtime with system properties (hence the metrics-ws.url is omitted
041    // in the properties)
042    Injector clientInjector = Guice.createInjector(
043      new UrlBindingModule(getBaseURI().toString() + CONTEXT, "metrics.ws.url"),
044      new MetricsWsClientModule(PropertiesUtil.loadProperties(PROPERTIES_FILE)));
045    wsClient = clientInjector.getInstance(CubeService.class);
046  }
047  
048  /**
049   * This simply does lookups to verify we can read something without error.
050   * This is an IT of the total wiring, and not intended to check any business logic of the cube.
051   */
052  @Ignore("Tried to connect to table outside Grizzly that no longer existed throwing TableNotFoundException")
053  public void basicLookup() {
054    wsClient.get(new ReadBuilder().at(OccurrenceCube.NUB_KEY, ANIMALIA_KEY));
055    wsClient.get(new ReadBuilder().at(OccurrenceCube.DATASET_KEY, UUID.randomUUID()));
056    wsClient.get(new ReadBuilder());
057  }
058  
059  /**
060   * An IT to simply check the scheme can be read, and that some rollups exist.
061   * Is not meant to test any business logic.
062   */
063  @Test
064  public void schema() {
065    assertTrue("CubeIo schema says no rollups which can't be true", 
066      wsClient.getSchema().size() > 0);
067  }
068}
