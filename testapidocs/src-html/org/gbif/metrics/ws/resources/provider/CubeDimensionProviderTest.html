001package org.gbif.metrics.ws.resources.provider;
002
003import org.gbif.api.model.metrics.cube.OccurrenceCube;
004import org.gbif.api.vocabulary.BasisOfRecord;
005
006import java.util.UUID;
007
008import javax.ws.rs.core.MultivaluedMap;
009
010import com.sun.jersey.core.util.MultivaluedMapImpl;
011import com.urbanairship.datacube.Address;
012import com.urbanairship.datacube.ReadBuilder;
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016
017/**
018 * Tests that the Jersey provider does it's job unpacking the HTTP params and constructing the internal datacube
019 * read builder properly.
020 */
021public class CubeDimensionProviderTest {
022
023  private final UUID DS = UUID.randomUUID();
024
025  // Builds an Address (package declarations necessary, as this is the INTERNAL one)
026  private Address getInternallyBuilt() {
027    return new ReadBuilder(org.gbif.metrics.cube.occurrence.OccurrenceCube.INSTANCE)
028      .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.BASIS_OF_RECORD, BasisOfRecord.OBSERVATION)
029      .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.NUB_KEY, 212)
030      .at(org.gbif.metrics.cube.occurrence.OccurrenceCube.DATASET_KEY, DS)
031      .build();
032  }
033
034  // An address built by the Jersey provider must be the same as a DC readbuilder version
035  @Test
036  public void testBuild() {
037    OccurrenceCubeReaderProvider b = new OccurrenceCubeReaderProvider(null);
038    MultivaluedMap<String, String> m = new MultivaluedMapImpl();
039    m.add(OccurrenceCube.BASIS_OF_RECORD.getKey(), BasisOfRecord.OBSERVATION.toString());
040    m.add(OccurrenceCube.NUB_KEY.getKey(), "212");
041    m.add(OccurrenceCube.DATASET_KEY.getKey(), DS.toString());
042    ReadBuilder rb = b.build(m);
043    assertEquals(getInternallyBuilt(), rb.build());
044  }
045
046  @Test(expected = IllegalArgumentException.class)
047  public void testFailureScenarios() {
048    OccurrenceCubeReaderProvider b = new OccurrenceCubeReaderProvider(null);
049    MultivaluedMap<String, String> m = new MultivaluedMapImpl();
050    m.add("nonesense", "should throw error");
051    b.build(m);
052  }
053
054}
