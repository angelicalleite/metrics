001package org.gbif.metrics.ws.resources;
002
003import java.util.Iterator;
004import java.util.Map;
005import java.util.UUID;
006
007import com.google.common.collect.Maps;
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011
012public class OccurrenceCubeResourceTest {
013
014  @Test
015  public void testSortDescending() {
016    OccurrenceCubeResource r = new OccurrenceCubeResource(null, null, null);
017    Map<CharSequence, Integer> m = Maps.newHashMap();
018    m.put("82ec8f46-f762-11e1-a439-00145eb45e9a", 2571); // same value
019    m.put("cf5ce014-6ac5-47fe-8b35-c70a9826562e", 2571); // same value
020    m.put("12ec8f46-f762-11e1-a439-00145eb45e9a", 100000);
021
022    Map<UUID, Integer> result = r.sortDescending(m);
023    assertEquals(3, result.size());
024    Iterator<Map.Entry<UUID, Integer>> iter = result.entrySet().iterator();
025    assertEquals("12ec8f46-f762-11e1-a439-00145eb45e9a", iter.next().getKey().toString());
026    assertEquals("82ec8f46-f762-11e1-a439-00145eb45e9a", iter.next().getKey().toString());
027    assertEquals("cf5ce014-6ac5-47fe-8b35-c70a9826562e", iter.next().getKey().toString());
028  }
029
030  @Test
031  public void testSortDescending2() {
032    OccurrenceCubeResource r = new OccurrenceCubeResource(null, null, null);
033    Map<Integer, Long> m = Maps.newHashMap();
034    m.put(1, 2571l); // same value
035    m.put(2, 2571l); // same value
036    m.put(3, 100000l);
037
038    Map<Integer, Long> result = r.sortDescendingValues(m);
039    assertEquals(3, result.size());
040    Iterator<Map.Entry<Integer, Long>> iter = result.entrySet().iterator();
041    assertEquals((Integer) 3, iter.next().getKey());
042    assertEquals((Integer) 1, iter.next().getKey());
043    assertEquals((Integer) 2, iter.next().getKey());
044  }
045}
