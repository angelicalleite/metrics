001package org.gbif.metrics.cube.mapred;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.api.vocabulary.EndpointType;
005
006import java.io.ByteArrayInputStream;
007import java.io.ByteArrayOutputStream;
008import java.io.DataInputStream;
009import java.io.DataOutput;
010import java.io.DataOutputStream;
011import java.io.IOException;
012
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertNull;
017import static org.junit.Assert.assertTrue;
018import static org.junit.Assert.fail;
019
020public class OccurrenceWritableTest {
021
022  @Test
023  public void testSerDe() {
024    OccurrenceWritable o =
025      new OccurrenceWritable(1, null, null, null, null, null, null, 1, false, null, "1", null, "GB", 0.89, null, 2012, 12,
026        BasisOfRecord.FOSSIL_SPECIMEN, EndpointType.DWC_ARCHIVE, 1);
027    ByteArrayOutputStream baos = new ByteArrayOutputStream();
028    DataOutput out = new DataOutputStream(baos);
029    try {
030      o.write(out);
031      ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
032      o = new OccurrenceWritable();
033      o.readFields(new DataInputStream(bais));
034      assertEquals(new Integer(1), o.getKingdomID());
035      assertEquals(new Integer(2012), o.getYear());
036      assertEquals(new Integer(12), o.getMonth());
037      assertEquals("GB", o.getHostCountryIsoCode());
038      assertNull(o.getPhylumID());
039      assertNull(o.getLongitude());
040      assertNull(o.getLongitude());
041      assertEquals(BasisOfRecord.FOSSIL_SPECIMEN, o.getBasisOfRecord());
042      assertEquals(EndpointType.DWC_ARCHIVE, o.getProtocol());
043
044    } catch (IOException e) {
045      fail(e.getMessage());
046    }
047  }
048
049  @Test
050  public void testCompareTo() {
051    OccurrenceWritable o1 =
052      new OccurrenceWritable(1, null, null, null, null, null, null, 1, false, null, "1", null, "GB", 0.89, null, 1, 12,
053        BasisOfRecord.FOSSIL_SPECIMEN, EndpointType.DWC_ARCHIVE, 1);
054    OccurrenceWritable o2 =
055      new OccurrenceWritable(1, null, null, null, null, null, null, 1, false, null, "1", null, "GB", 0.89, null, 1, 12,
056        BasisOfRecord.FOSSIL_SPECIMEN, EndpointType.DWC_ARCHIVE, 1);
057    OccurrenceWritable o3 =
058      new OccurrenceWritable(1, null, null, 10, null, null, null, 1, false, null, "1", "GB", null, 0.89, null, 1, 12,
059        BasisOfRecord.FOSSIL_SPECIMEN, EndpointType.DWC_ARCHIVE, 1);
060    assertEquals(0, o1.compareTo(o2));
061    assertTrue(o1.compareTo(o3) > 0);
062    assertTrue(o3.compareTo(o2) < 0);
063  }
064}
