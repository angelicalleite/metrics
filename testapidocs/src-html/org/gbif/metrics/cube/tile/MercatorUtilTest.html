001package org.gbif.metrics.cube.tile;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007
008public class MercatorUtilTest {
009
010  @Test
011  public void testGetOffset() {
012    assertEquals(92, MercatorProjectionUtil.getOffsetY(45d, -90d, 0));
013    assertEquals(128, MercatorProjectionUtil.getOffsetX(0d, 0d, 0));
014    assertEquals(128, MercatorProjectionUtil.getOffsetY(0d, 0d, 0));
015    assertEquals(0, MercatorProjectionUtil.getOffsetX(0d, 0d, 1));
016    assertEquals(0, MercatorProjectionUtil.getOffsetY(0d, 0d, 1));
017    assertEquals(0, MercatorProjectionUtil.getOffsetX(0d, 0d, 2));
018    assertEquals(0, MercatorProjectionUtil.getOffsetY(0d, 0d, 2));
019
020    // Canberra, AU
021    double lng = 149.1;
022    double lat = -35.2;
023    assertEquals(0, MercatorProjectionUtil.toTileX(lng, 0));
024    assertEquals(1, MercatorProjectionUtil.toTileX(lng, 1));
025    assertEquals(3, MercatorProjectionUtil.toTileX(lng, 2));
026    assertEquals(0, MercatorProjectionUtil.toTileY(lat, 0));
027    assertEquals(1, MercatorProjectionUtil.toTileY(lat, 1));
028    assertEquals(2, MercatorProjectionUtil.toTileY(lat, 2));
029  }
030}
