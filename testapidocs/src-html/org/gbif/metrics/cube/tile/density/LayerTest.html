001package org.gbif.metrics.cube.tile.density;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.metrics.cube.mapred.OccurrenceWritable;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009
010
011public class LayerTest {
012
013  @Test
014  public void testInferFrom() {
015    OccurrenceWritable o = new OccurrenceWritable();
016    o.setBasisOfRecord(BasisOfRecord.LIVING_SPECIMEN);
017    assertEquals(Layer.LIVING, Layer.inferFrom(o));
018    o.setBasisOfRecord(BasisOfRecord.HUMAN_OBSERVATION);
019    assertEquals(Layer.OBS_NO_YEAR, Layer.inferFrom(o));
020    o.setYear(1800);
021    assertEquals(Layer.OBS_PRE_1900, Layer.inferFrom(o));
022    o.setYear(1900);
023    assertEquals(Layer.OBS_1900_1910, Layer.inferFrom(o));
024    o.setYear(1920);
025    assertEquals(Layer.OBS_1920_1930, Layer.inferFrom(o));
026    o.setYear(1929);
027    assertEquals(Layer.OBS_1920_1930, Layer.inferFrom(o));
028    o.setYear(2012);
029    assertEquals(Layer.OBS_2010_2020, Layer.inferFrom(o));
030    o.setBasisOfRecord(BasisOfRecord.OBSERVATION);
031    assertEquals(Layer.OBS_2010_2020, Layer.inferFrom(o));
032    o.setYear(2020);
033    assertEquals(Layer.OBS_NO_YEAR, Layer.inferFrom(o));
034  }
035}
