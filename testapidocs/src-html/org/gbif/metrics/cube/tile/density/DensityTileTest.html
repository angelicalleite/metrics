033  @Test
034  public void testCell() {
035
036    DensityTile tile =
037      DensityTile.builder(0, 0, 0, 64)
038        // Z, X, Y, ClusterSize
039        .collect(Layer.SP_NO_YEAR, -0.1d, 0.1d, 1)
040        // lat, lng, count
041        .collect(Layer.SP_2000_2010, -0.1d, 0.1d, 1000).collect(Layer.OBS_PRE_1900, -0.1d, 0.1d, 1)
042        .collect(Layer.SP_NO_YEAR, -0.1d, 0.1d, 2).build();
043
044    assertEquals(3, tile.cell(Layer.SP_NO_YEAR, 2, 2));
045    assertEquals(0, tile.cell(Layer.SP_NO_YEAR, 0, 0));
046    assertEquals(1, tile.cell(Layer.OBS_PRE_1900, 2, 2));
047    assertEquals(1000, tile.cell(Layer.SP_2000_2010, 2, 2));
048    assertEquals(3, tile.layers().size());
049
050    try {
051      byte[] b = tile.serialize();
052      tile = DensityTile.DESERIALIZE(b);
053      assertEquals(3, tile.cell(Layer.SP_NO_YEAR, 2, 2));
054      assertEquals(0, tile.cell(Layer.SP_NO_YEAR, 0, 0));
055      assertEquals(1, tile.cell(Layer.OBS_PRE_1900, 2, 2));
056      assertEquals(1000, tile.cell(Layer.SP_2000_2010, 2, 2));
057      assertEquals(3, tile.layers().size());
058    } catch (IOException e) {
059      fail(e.getMessage());
060    }
061  }

063  @Test
064  public void testDownscale() {
065    Map<Layer, Map<Integer, Integer>> layers =
066      ImmutableMap.<Layer, Map<Integer, Integer>>builder()
067        .put(Layer.FOSSIL, ImmutableMap.<Integer, Integer>of(514, 1, 515, 1))
068        .build();
069    DensityTile tile1px = new DensityTile(0, 0, 0, 1, layers);
070
071    DensityTile tile2px = tile1px.downscale(2);
072    assertEquals("Should still have only 1 layer", 1, tile2px.layers().size());
073    assertEquals("Downsizing should have merged adjacent cells", 1, tile2px.layers().get(Layer.FOSSIL).size());
074    assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
075      tile2px.layers().get(Layer.FOSSIL).get(129)); // the new key
076
077    DensityTile tile4px = tile1px.downscale(4);
078    assertEquals("Should still have only 1 layer", 1, tile4px.layers().size());
079    assertEquals("Downsizing should have merged adjacent cells", 1, tile4px.layers().get(Layer.FOSSIL).size());
080    assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
081      tile4px.layers().get(Layer.FOSSIL).get(0)); // the new key
082
083    // test the 2-4 downscale
084    tile4px = tile2px.downscale(4);
085    assertEquals("Should still have only 1 layer", 1, tile4px.layers().size());
086    assertEquals("Downsizing should have merged adjacent cells", 1, tile4px.layers().get(Layer.FOSSIL).size());
087    assertEquals("Downsizing should have accumulated adjacent counts", Integer.valueOf(2),
088      tile4px.layers().get(Layer.FOSSIL).get(0)); // the new key
089  }

130  private final int totalCount(DensityTile tile) {
131    int total = 0;
132    for (Entry<Layer, Map<Integer, Integer>> e : tile.layers().entrySet()) {
133      for (Integer count : e.getValue().values()) {
134        total += count;
135      }
136    }
137    return total;
138  }
