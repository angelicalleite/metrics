001package org.gbif.metrics.cube.index.common;
002
003import org.gbif.metrics.cube.index.common.UuidIntMap.CountByDatasetDeserializer;
004
005import java.util.HashMap;
006import java.util.Map;
007import java.util.UUID;
008
009import org.junit.Test;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNotNull;
013
014/**
015 * Basic tests to verify the counts increment correctly and SerDe works ok.
016 */
017public class UuidIntMapTest {
018
019  @Test
020  public void testIncrements() {
021    Map<CharSequence, Integer> m = new HashMap<CharSequence, Integer>();
022    m.put("1", 1);
023    UuidIntMap o1 = new UuidIntMap(m);
024    UuidIntMap o2 = new UuidIntMap(m);
025
026    // basic addition
027    UuidIntMap o3 = o1.add(o2);
028    assertNotNull(o3);
029    assertNotNull(o3.getCounts());
030    assertEquals(1, o3.getCounts().size());
031    assertEquals(2, (int) o3.getCounts().get("1"));
032
033    // basic subtraction
034    UuidIntMap o4 = o3.subtract(o2);
035    assertNotNull(o4);
036    assertNotNull(o4.getCounts());
037    assertEquals(1, o4.getCounts().size());
038    assertEquals(1, (int) o4.getCounts().get("1"));
039    o4 = o4.subtract(o2);
040    assertNotNull(o4);
041    assertNotNull(o4.getCounts());
042    assertEquals(1, o4.getCounts().size());
043    assertEquals(0, (int) o4.getCounts().get("1"));
044    o4 = o4.subtract(o2);
045    assertNotNull(o4);
046    assertNotNull(o4.getCounts());
047    assertEquals(1, o4.getCounts().size());
048    assertEquals(0, (int) o4.getCounts().get("1")); // ensure does not go negative
049  }
050
051
052  @Test
053  public void testIncrementsWithUUIDs() {
054    UUID randomUuid = UUID.randomUUID();
055    Map<UUID, Integer> m = new HashMap<UUID, Integer>();
056    m.put(randomUuid, 1);
057    UuidIntMap o1 = UuidIntMap.newInstance(m);
058    UuidIntMap o2 = UuidIntMap.DESERIALIZER.fromBytes(o1.serialize());
059
060    // basic addition
061    UuidIntMap o3 = o1.add(o2);
062    assertNotNull(o3);
063    assertNotNull(o3.getCounts());
064    assertEquals(1, o3.getCounts().size());
065    assertEquals(2, (int) o3.getCounts().get(randomUuid.toString()));
066
067    // basic subtraction
068    UuidIntMap o4 = o3.subtract(o2);
069    assertNotNull(o4);
070    assertNotNull(o4.getCounts());
071    assertEquals(1, o4.getCounts().size());
072    assertEquals(1, (int) o4.getCounts().get(randomUuid.toString()));
073    o4 = o4.subtract(o2);
074    assertNotNull(o4);
075    assertNotNull(o4.getCounts());
076    assertEquals(1, o4.getCounts().size());
077    assertEquals(0, (int) o4.getCounts().get(randomUuid.toString()));
078    o4 = o4.subtract(o2);
079    assertNotNull(o4);
080    assertNotNull(o4.getCounts());
081    assertEquals(1, o4.getCounts().size());
082    assertEquals(0, (int) o4.getCounts().get(randomUuid.toString())); // ensure does not go negative
083  }
084
085  @Test
086  public void testSerDe() {
087    Map<CharSequence, Integer> m = new HashMap<CharSequence, Integer>();
088    m.put("1", 1);
089    m.put("2", 2);
090    UuidIntMap o1 = new UuidIntMap(m);
091    byte[] b = o1.serialize();
092    UuidIntMap o2 = new CountByDatasetDeserializer().fromBytes(b);
093    assertEquals(o1.hashCode(), o2.hashCode());
094    assertEquals(o1, o2);
095  }
