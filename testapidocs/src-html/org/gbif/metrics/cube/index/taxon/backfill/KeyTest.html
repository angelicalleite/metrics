001package org.gbif.metrics.cube.index.taxon.backfill;
002
003import java.io.ByteArrayInputStream;
004import java.io.DataInputStream;
005import java.io.DataOutputStream;
006import java.io.IOException;
007import java.util.UUID;
008
009import org.apache.commons.io.output.ByteArrayOutputStream;
010import org.junit.Test;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.fail;
014
015
016public class KeyTest {
017
018  @Test
019  public void test() {
020    UUID u = UUID.randomUUID();
021    Key k = new Key(u, 1);
022    ByteArrayOutputStream baos = new ByteArrayOutputStream();
023    DataOutputStream dos = new DataOutputStream(baos);
024    try {
025      k.write(dos);
026      dos.flush();
027      byte[] bytes = baos.toByteArray();
028      Key k2 = new Key();
029      k2.readFields(new DataInputStream(new ByteArrayInputStream(bytes)));
030      assertEquals(k, k2);
031    } catch (IOException e) {
032      e.printStackTrace();
033      fail(e.getMessage());
034    }
035    
036  }
037
038}
