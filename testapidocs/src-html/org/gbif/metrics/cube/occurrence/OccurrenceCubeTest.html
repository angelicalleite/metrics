033/**
034 * This is a test of the general functionality for the occurrence cube, the rollups
035 * and inherently includes verification that the normalization provided by {@link OccurrenceAddressUtil} operates as
036 * expected.
037 */
038public class OccurrenceCubeTest {
039
040  private DataCubeIo<LongOp> cubeIo;
041
042  // Sets up the cube
043  @Before
044  public void setup() {
045    ConcurrentMap<BoxedByteArray, byte[]> backingMap = Maps.newConcurrentMap();
046    IdService idService = new CachingIdService(4, new MapIdService(), "id");
047    DbHarness<LongOp> dbHarness =
048      new MapDbHarness<LongOp>(backingMap, LongOp.DESERIALIZER, CommitType.READ_COMBINE_CAS, idService);
049    cubeIo = new DataCubeIo<LongOp>(OccurrenceCube.INSTANCE, dbHarness, 1, Long.MAX_VALUE, SyncLevel.FULL_SYNC);
050  }
051
052
053  @Test
054  public void testCompleteness() throws InterruptedException, AsyncException, IOException {
055    for (Dimension<?> dim : org.gbif.api.model.metrics.cube.OccurrenceCube.DIMENSIONS) {
056      assertTrue("OccurrenceCube.API_MAPPING missing API dimension " + dim.getKey(),
057        OccurrenceCube.API_MAPPING.containsKey(dim));
058    }
059  }
060
061  @Test
062  public void testRollups() throws InterruptedException, AsyncException, IOException {
063    final UUID ds1 = UUID.randomUUID();
064    final UUID ds2 = UUID.randomUUID();
065    final Occurrence o1 =
066      occurrenceOf(123456, 1, 2, 3, 4, 5, 6, 7, 7, ds1, BasisOfRecord.OBSERVATION, Country.UKRAINE, 2012, 10.0, 11.0,
067        EndpointType.BIOCASE);
068    final Occurrence o2 =
069      occurrenceOf(123457, 1, 2, 3, 4, 5, 6, 7, 7, ds1, BasisOfRecord.PRESERVED_SPECIMEN, Country.UKRAINE, 2012, 10.0,
070        11.0, EndpointType.DWC_ARCHIVE);
071    final Occurrence o3 =
072      occurrenceOf(123458, 1, 2, 3, 4, 5, 6, 8, 9, ds2, BasisOfRecord.PRESERVED_SPECIMEN, Country.UKRAINE, 2012, 10.0,
073        11.0, EndpointType.DIGIR);
074
075    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o1, new LongOp(1)));
076    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o2, new LongOp(1)));
077    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o3, new LongOp(1)));
078    cubeIo.flush();
079
080    // Tests that NUB KEY are correctly normalized
081    Assert.assertEquals(3L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 1)));
082    Assert.assertEquals(2L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 7)));
083    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 8)));
084    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 9)));
085
086    // Assertion that normalization of taxa don't incorrectly screw counts (http://dev.gbif.org/issues/browse/MET-11)
087    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.DATASET_KEY, ds2)));
088
089    // Random checking
090    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.DATASET_KEY, ds2)));
091    Assert.assertEquals(
092      0L,
093      getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.BASIS_OF_RECORD,
094        BasisOfRecord.FOSSIL_SPECIMEN)
095        .at(OccurrenceCube.HOST_COUNTRY, Country.AFGHANISTAN)));
096  }
097
098  @Test
099  public void testSubtraction() throws InterruptedException, AsyncException, IOException {
100    final UUID ds1 = UUID.randomUUID();
101    final Occurrence o1 =
102      occurrenceOf(123456, 1, 2, 3, 4, 5, 6, 7, 7, ds1, BasisOfRecord.OBSERVATION, Country.UKRAINE, 2012, 10.0, 11.0,
103        EndpointType.DWC_ARCHIVE);
104
105    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o1, new LongOp(1)));
106    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o1, new LongOp(1)));
107    cubeIo.flush();
108    Assert.assertEquals(2L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 1)));
109    Assert.assertEquals(2L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.DATASET_KEY, ds1)));
110
111    cubeIo.writeAsync(OccurrenceAddressUtil.cubeMutation(o1, new LongOp(-1)));
112    cubeIo.flush();
113    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.NUB_KEY, 1)));
114    Assert.assertEquals(1L, getCount(new ReadBuilder(OccurrenceCube.INSTANCE).at(OccurrenceCube.DATASET_KEY, ds1)));
115  }
116
117  private long getCount(ReadBuilder r) throws IOException, InterruptedException {
118    Optional<LongOp> opt = cubeIo.get(r);
119    if (!opt.isPresent()) {
120      return 0L;
121    } else {
122      return opt.get().getLong();
123    }
124  }
125
126  private Occurrence occurrenceOf(
127    Integer key, Integer kingdom, Integer phylum, Integer classs, Integer order, Integer family, Integer genus,
128    Integer species, Integer nub,
129    UUID dataset, BasisOfRecord bor, Country country, Integer year, Double latitude, Double longitude,
130    EndpointType protocol) {
131    Occurrence occ = new Occurrence();
132    occ.setKey(key);
133    occ.setKingdomKey(kingdom);
134    occ.setPhylumKey(phylum);
135    occ.setClassKey(classs);
136    occ.setOrderKey(order);
137    occ.setFamilyKey(family);
138    occ.setGenusKey(genus);
139    occ.setSpeciesKey(species);
140    occ.setTaxonKey(nub);
141    occ.setDatasetKey(dataset);
142    occ.setBasisOfRecord(bor);
143    occ.setCountry(country);
144    occ.setYear(year);
145    occ.setLatitude(latitude);
146    occ.setLongitude(longitude);
147    occ.setProtocol(protocol);
148
149    return occ;
150  }
